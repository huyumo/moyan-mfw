## 6.1 日志分级

### 6.1.1 日志级别定义

| 级别 | 说明 | 使用场景 | 示例 |
|------|------|----------|------|
| `ERROR` | 错误 | 需要立即处理的错误 | 数据库连接失败、第三方服务异常 |
| `WARN` | 警告 | 潜在问题，不影响当前请求 | 参数已废弃、缓存失效 |
| `INFO` | 信息 | 重要的系统事件 | 用户登录、配置加载、服务启动 |
| `DEBUG` | 调试 | 开发调试信息 | 请求参数、SQL 语句、处理流程 |
| `TRACE` | 追踪 | 详细的执行轨迹 | 方法入口出口、变量值变化 |

### 6.1.2 日志工具实现

```typescript
// utils/logger.ts
import winston from 'winston';

const logger = winston.createLogger({
  level: process.env.LOG_LEVEL || 'info',
  format: winston.format.combine(
    winston.format.timestamp({ format: 'YYYY-MM-DD HH:mm:ss' }),
    winston.format.errors({ stack: true }),
    winston.format.splat(),
    winston.format.json(),
  ),
  defaultMeta: {
    service: 'moyan-mfw-backend',
    version: process.env.npm_package_version,
  },
  transports: [
    // 错误日志单独输出
    new winston.transports.File({
      filename: 'logs/error.log',
      level: 'error',
      maxsize: 10485760,  // 10MB
      maxFiles: 7,        // 保留 7 天
    }),
    // 所有日志
    new winston.transports.File({
      filename: 'logs/combined.log',
      maxsize: 10485760,
      maxFiles: 7,
    }),
  ],
});

// 开发环境输出到控制台
if (process.env.NODE_ENV !== 'production') {
  logger.add(new winston.transports.Console({
    format: winston.format.combine(
      winston.format.colorize(),
      winston.format.simple(),
    ),
  }));
}

export default logger;
```

### 6.1.3 日志使用规范

```typescript
// ✅ 推荐：结构化日志
logger.info('用户登录成功', {
  userId: user.id,
  username: user.username,
  ip: request.ip,
  userAgent: request.headers['user-agent'],
});

logger.error('数据库查询失败', {
  error: error.message,
  stack: error.stack,
  query: 'SELECT * FROM users WHERE id = ?',
  params: [userId],
});

// ❌ 避免：无上下文日志
console.log('出错了');
logger.info('登录');
```

### 6.1.4 请求日志中间件

```typescript
// middleware/logger.middleware.ts
@Injectable()
export class LoggerMiddleware implements NestMiddleware {
  private readonly logger = new Logger('HTTP');

  use(req: Request, res: Response, next: NextFunction) {
    const { method, originalUrl } = req;
    const startTime = Date.now();

    res.on('finish', () => {
      const { statusCode } = res;
      const contentLength = res.get('content-length');
      const duration = Date.now() - startTime;

      this.logger.log(
        statusCode >= 500 ? 'error' : 'info',
        `${method} ${originalUrl}`,
        {
          statusCode,
          contentLength,
          duration: `${duration}ms`,
          userAgent: req.get('user-agent'),
          ip: req.ip,
        },
      );
    });

    next();
  }
}
```

## 6.2 审计日志

### 6.2.1 审计日志表结构

```sql
CREATE TABLE sys_audit_log (
  id BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '主键 ID',
  user_id VARCHAR(36) NOT NULL COMMENT '操作用户 ID',
  username VARCHAR(50) NOT NULL COMMENT '操作用户名',
  action VARCHAR(100) NOT NULL COMMENT '操作类型',
  module VARCHAR(50) NOT NULL COMMENT '所属模块',
  resource_type VARCHAR(50) COMMENT '资源类型',
  resource_id VARCHAR(36) COMMENT '资源 ID',
  request_method VARCHAR(10) COMMENT '请求方法',
  request_url VARCHAR(500) COMMENT '请求 URL',
  request_body JSON COMMENT '请求体',
  response_status INT COMMENT '响应状态码',
  response_body JSON COMMENT '响应体',
  client_ip VARCHAR(50) COMMENT '客户端 IP',
  user_agent VARCHAR(500) COMMENT '用户代理',
  execution_time INT COMMENT '执行时间 (ms)',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP COMMENT '操作时间',
  INDEX idx_user_id (user_id),
  INDEX idx_action (action),
  INDEX idx_created_at (created_at),
  INDEX idx_module (module)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='审计日志表';
```

### 6.2.2 审计日志装饰器

```typescript
// decorators/audit-log.decorator.ts
export const AuditLog = (options: {
  action: string;
  module: string;
  resourceType?: string;

  return SetMetadata('audit-log', options);
};

// 使用

export class UserController {
  @Post()
  @AuditLog({ action: 'CREATE_USER', module: 'SYSTEM', resourceType: 'USER' })
  async createUser(@Body() dto: CreateUserDTO) {
    // ...
  }

  @Delete(':id')
  @AuditLog({ action: 'DELETE_USER', module: 'SYSTEM', resourceType: 'USER' })
  async deleteUser(@Param('id') id: string) {
    // ...
  }
}
```

### 6.2.3 审计日志拦截器

```typescript
// interceptors/audit-log.interceptor.ts
@Injectable()
export class AuditLogInterceptor implements NestInterceptor {
  constructor(
    private reflector: Reflector,
    private auditLogService: AuditLogService,
  ) {}

  async intercept(
    context: ExecutionContext,
    next: CallHandler,
  ): Promise<Observable<any>> {
    const auditLogMeta = this.reflector.get('audit-log', context.getHandler());

    if (!auditLogMeta) {
      return next.handle();
    }

    const request = context.switchToHttp().getRequest<Request>();
    const startTime = Date.now();

    return next.handle().pipe(
      tap(async (response) => {
        await this.auditLogService.log({
          ...auditLogMeta,
          userId: request.user?.sub,
          username: request.user?.username,
          requestMethod: request.method,
          requestUrl: request.originalUrl,
          requestBody: request.body,
          responseStatus: response?.code === 0 ? 200 : 400,
          responseBody: response,
          clientIp: request.ip,
          userAgent: request.headers['user-agent'],
          executionTime: Date.now() - startTime,
        });
      }),
    );
  }
}
```

## 6.3 监控指标

### 6.3.1 应用健康检查

```typescript
// controllers/health.controller.ts
@Controller('health')
export class HealthController {
  constructor(
    @Inject('DB_CONNECTION')
    private dbConnection: Connection,
  ) {}

  @Get()
  async health() {
    return {
      status: 'ok',
      timestamp: new Date().toISOString(),
      uptime: process.uptime(),
    };
  }

  @Get('ready')
  async ready() {
    // 检查依赖服务是否就绪
    try {
      await this.dbConnection.query('SELECT 1');
      return { status: 'ok' };
    } catch (error) {
      throw new ServiceUnavailableException('Database not ready');
    }
  }

  @Get('live')
  async live() {
    // 存活检查，始终返回成功
    return { status: 'ok' };
  }
}
```

### 6.3.2 性能监控指标

| 指标 | 说明 | 告警阈值 |
|------|------|----------|
| HTTP 请求量 | 每秒请求数 (RPS) | > 1000 |
| 响应时间 | P50/P95/P99 延迟 | P99 > 2s |
| 错误率 | 5xx 错误占比 | > 1% |
| CPU 使用率 | 服务器 CPU 使用率 | > 80% |
| 内存使用率 | 服务器内存使用率 | > 85% |
| 数据库连接数 | 活跃连接数 | > 80% 最大连接 |
| 慢查询数 | 执行时间 > 1s 的查询 | > 10/min |

## 6.4 告警通知

### 6.4.1 告警规则配置

```yaml
# alerting.yml
rules:
  - name: HighErrorRate
    condition: error_rate > 0.01  # 错误率 > 1%
    duration: 5m                   # 持续 5 分钟
    severity: critical
    channels:
      - slack
      - email
      - sms

  - name: HighLatency
    condition: p99_latency > 2s
    duration: 5m
    severity: warning
    channels:
      - slack
      - email

  - name: ServiceDown
    condition: health_check == false
    duration: 1m
    severity: critical
    channels:
      - slack
      - email
      - sms
      - phone
```

### 6.4.2 告警通知模板

```typescript
// 告警消息模板
interface AlertMessage {
  alertName: string;
  severity: 'critical' | 'warning' | 'info';
  startTime: string;
  duration: string;
  currentValue: string;
  threshold: string;
  affectedService: string;
  runbook: string;  // 处理手册链接
}

// Slack 通知
async function sendSlackAlert(alert: AlertMessage) {
  const color = alert.severity === 'critical' ? 'danger' : 'warning';

  await slack.chat.postMessage({
    channel: '#alerts',
    attachments: [{
      color,
      title: `🚨 ${alert.alertName}`,
      fields: [
        { title: '服务', value: alert.affectedService, short: true },
        { title: '严重程度', value: alert.severity, short: true },
        { title: '当前值', value: alert.currentValue, short: true },
        { title: '阈值', value: alert.threshold, short: true },
        { title: '持续时间', value: alert.duration, short: true },
        { title: '处理手册', value: `<${alert.runbook}|点击查看>`, short: true },
      ],
    }],
  });
}
```
