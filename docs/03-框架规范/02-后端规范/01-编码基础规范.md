## 1.1 命名规范

### 1.1.1 文件命名

| 类型 | 规则 | 示例 | 等级 |
|------|------|------|------|
| 服务类 | PascalCase + Service | `UserService.ts`, `OrderService.ts` | 🔴 |
| 控制器 | PascalCase + Controller | `UserController.ts`, `AuthController.ts` | 🔴 |
| 数据模型 | PascalCase + Model | `UserModel.ts`, `RoleModel.ts` | 🔴 |
| 数据传输对象 | PascalCase + DTO | `CreateUserDTO.ts`, `UserResponseDTO.ts` | 🔴 |
| 数据访问对象 | PascalCase + DAO | `UserDAO.ts`, `RoleDAO.ts` | 🔴 |
| 接口定义 | PascalCase + Service/I + 名词 | `IUserService.ts` 或 `UserServiceInterface.ts` | 🟢 |
| 枚举类型 | PascalCase + Enum | `UserStatusEnum.ts`, `RoleTypeEnum.ts` | 🟢 |
| 工具函数 | camelCase | `utils.ts`, `formatters.ts`, `validators.ts` | 🔴 |
| 类型定义 | camelCase + types | `user-types.ts`, `common-types.ts` | 🟢 |
| 配置文件 | camelCase + config | `database.config.ts`, `app.config.ts` | 🟢 |
| 中间件 | camelCase + middleware | `auth.middleware.ts`, `logger.middleware.ts` | 🟢 |
| 常量文件 | UPPER_SNAKE_CASE | `ERROR_CODES.ts`, `CONSTANTS.ts` | 🟢 |

### 1.1.2 目录命名

```
✅ 推荐：
src/controllers/          # 控制器层
src/services/             # 服务层
src/models/               # 数据模型层
src/dao/                  # 数据访问层
src/dto/                  # 数据传输对象
src/middleware/           # 中间件
src/utils/                # 工具函数
src/config/               # 配置文件
src/types/                # 类型定义
src/enums/                # 枚举类型
src/interfaces/           # 接口定义

❌ 避免：
src/Controller/           # 不要大写
src/controller/           # 不要单数（除非语义需要）
src/myUtils/              # 避免个人化命名
```

```typescript
// 变量 - camelCase
const userName = 'test';
const isLoading = false;
const userList: User[] = [];

// 常量 - UPPER_SNAKE_CASE
const MAX_RETRY_COUNT = 3;
const API_TIMEOUT = 10000;
const DEFAULT_PAGE_SIZE = 10;

// 类/接口 - PascalCase
class UserService {
  async getUserById(id: string): Promise<User> {
    // ...
  }
}

interface UserInfo {
  id: string;
  name: string;
  email: string;
}

// 枚举 - PascalCase + 成员 UPPER_SNAKE_CASE
enum UserStatus {
  ACTIVE = 'active',
  INACTIVE = 'inactive',
  DISABLED = 'disabled',
}

// 泛型类型 - PascalCase + T 后缀
interface ResponseT<T> {
  code: number;
  data: T;

}
```

## 1.4 导入导出规范

### 1.4.1 导入顺序

```typescript
// 第 1 部分：Node.js 内置模块
import { join, resolve } from 'path';
import { readFileSync } from 'fs';

// 第 2 部分：第三方模块
import express from 'express';
import { Connection } from 'typeorm';

// 第 3 部分：内部模块（绝对路径）
import { UserService } from '@/services/UserService';
import { User } from '@/models/User';

// 第 4 部分：相对路径
import { IUser } from '../interfaces/IUser';
import { formatUser } from '../utils/formatters';

// 第 5 部分：类型导入（TypeScript）
import type { Request, Response } from 'express';
import type { UserDTO } from '../dto/UserDTO';
```

### 1.4.2 导入格式

```typescript
// ✅ 推荐：
// 单个导出 - 使用花括号
import { UserService } from '@/services/UserService';

// 默认导出 - 不使用花括号
import express from 'express';

// 多个导出 - 按字母顺序排列
import {
  AuthService,

  RoleService,

// ❌ 避免：
// 不要在一条语句中混用默认导出和命名导出

// 不要有未使用的导入
import { UnusedService } from '@/services/UnusedService';  // ❌
```

## 1.6 错误处理规范

### 1.6.1 自定义错误类

```typescript
// ✅ 推荐：
// 定义业务错误类
export class UserNotFoundError extends Error {
  constructor(userId: string) {
    super(`用户 ${userId} 不存在`);
    this.name = 'UserNotFoundError';
  }
}

export class InvalidCredentialsError extends Error {
  constructor() {
    super('用户名或密码错误');
    this.name = 'InvalidCredentialsError';
  }
}

// 使用
async getUser(id: string) {
  const user = await this.userDAO.findById(id);
  if (!user) {
    throw new UserNotFoundError(id);
  }
  return user;
}
```

### 1.6.2 Try-Catch 规范

```typescript
// ✅ 推荐：
// 捕获具体错误类型
async function handleRequest() {
  try {
    await userService.process();
  } catch (error) {
    if (error instanceof UserNotFoundError) {
      // 处理用户不存在
      logger.warn('用户不存在:', error.message);
    } else if (error instanceof DatabaseError) {
      // 处理数据库错误
      logger.error('数据库错误:', error);
      throw error;
    } else {
      // 未知错误
      logger.error('未知错误:', error);
      throw error;
    }
  }
}

// ❌ 避免：
// 不要捕获不处理的异常
async function handleRequest() {
  try {
    await userService.process();
  } catch (error) {
    // 空 catch 块
  }
}
```

## 1.7 异步编程规范

### 1.7.1 Promise 使用

```typescript
// ✅ 推荐：
// 使用 async/await
async function getUserList() {
  const users = await userDAO.findAll();
  return users;
}

// 并发处理使用 Promise.all
async function getUserWithRoles(userId: string) {
  const [user, roles] = await Promise.all([
    userDAO.findById(userId),
    roleDAO.findByUserId(userId),
  ]);
  return { user, roles };
}

// ❌ 避免：
// 不要使用.then() 链式调用
userDAO.findById(id).then(user => {
  return roleDAO.findByUserId(user.id).then(roles => {
    return { user, roles };
  });
});
```

### 1.7.2 超时处理

```typescript
// ✅ 推荐：
// 为异步操作添加超时
async function fetchWithTimeout(
  url: string,
  timeout: number = 5000
) {
  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(), timeout);

  try {
    const response = await fetch(url, {
      signal: controller.signal,
    });
    return response;
  } finally {
    clearTimeout(timeoutId);
  }
}
```

## 1.8 安全规范

### 1.8.1 密码处理

```typescript
// ✅ 推荐：
// 使用 bcrypt 加密密码
import bcrypt from 'bcrypt';

const SALT_ROUNDS = 10;

async function hashPassword(password: string): Promise<string> {
  return bcrypt.hash(password, SALT_ROUNDS);
}

async function verifyPassword(
  password: string,
  hash: string
): Promise<boolean> {
  return bcrypt.compare(password, hash);
}

// ❌ 避免：
// 不要明文存储密码
// 不要使用 MD5/SHA1 等弱哈希算法
```

### 1.8.2 SQL 注入防护

```typescript
// ✅ 推荐：
// 使用参数化查询
async function getUserByName(name: string) {
  return this.userRepository
    .createQueryBuilder('user')
    .where('user.name = :name', { name })
    .getOne();
}

// ❌ 避免：
// 不要拼接 SQL 字符串
async function getUserByName(name: string) {
  // 危险！可能被 SQL 注入
  return this.query(`SELECT * FROM users WHERE name = '${name}'`);
}
```

### 1.8.3 敏感数据脱敏

```typescript
// ✅ 推荐：
// 返回前脱敏敏感数据
function sanitizeUser(user: User): UserResponseDTO {
  return {
    id: user.id,
    name: user.name,
    email: user.email,
    phone: user.phone.replace(/(\d{3})\d{4}(\d{4})/, '$1****$2'),
    // 不返回密码哈希
  };
}

// ❌ 避免：
// 不要直接返回包含敏感数据的对象
return user;  // 包含 passwordHash 等敏感字段
```
