**MySQL TestContainer 配置：**

```typescript
// test/database.ts
import { MySqlContainer, StartedMySqlContainer } from '@testcontainers/mysql';
import { createPool, Pool } from 'mysql2/promise';

let container: StartedMySqlContainer;
let pool: Pool;

export async function setupTestDatabase() {
  // 启动 MySQL 容器
  container = await new MySqlContainer()
    .withImage('mysql:8.0')
    .withRootPassword('test-root-password')
    .withDatabase('test_db')
    .withUsername('test_user')
    .withPassword('test_password')
    .withExposedPorts(3306)
    .start();

  // 创建连接池
  pool = createPool({
    host: container.getHost(),
    port: container.getPort(),
    user: container.getUsername(),
    password: container.getPassword(),
    database: container.getDatabase(),
    waitForConnections: true,
    connectionLimit: 5,
  });

  // 执行数据库迁移
  await runMigrations(pool);

  return pool;
}

export async function teardownTestDatabase() {
  if (pool) {
    await pool.end();
  }
  if (container) {
    await container.stop();
  }
}

export async function clearDatabase() {
  // 清空所有表数据（保留表结构）
  await pool.query('SET FOREIGN_KEY_CHECKS = 0');
  const tables = await pool.query(
    "SELECT table_name FROM information_schema.tables WHERE table_schema = 'test_db'"
  );
  for (const table of tables[0] as any[]) {
    await pool.query(`TRUNCATE TABLE \`${table.table_name}\``);
  }
  await pool.query('SET FOREIGN_KEY_CHECKS = 1');
}

async function runMigrations(pool: Pool) {
  // 执行迁移脚本
  const migrationFiles = ['001-create-users.sql', '002-create-roles.sql'];
  for (const file of migrationFiles) {
    const sql = await fs.readFile(`test/migrations/${file}`, 'utf-8');
    await pool.query(sql);
  }
}
```

import request from 'supertest';
import { app } from '../../src/main';

describe('User API Integration', () => {

        email: 'test@example.com',
      });

    expect(registerRes.status).toBe(201);
    expect(registerRes.body.code).toBe(0);
    testUser = registerRes.body.data;

      .post('/api/v1/auth/login')

    authToken = loginRes.body.data.accessToken;
  });

  describe('POST /api/v1/users', () => {
    it('应该成功创建用户', async () => {
      const res = await request(app)
        .post('/api/v1/users')
        .set('Authorization', `Bearer ${authToken}`)
        .send({
          username: 'newuser',
          password: 'password123',
          nickname: '测试用户',
          email: 'newuser@example.com',
        });

      expect(res.status).toBe(201);
      expect(res.body.code).toBe(0);
      expect(res.body.data.username).toBe('newuser');
      expect(res.body.data).toHaveProperty('id');
      expect(res.body.data).not.toHaveProperty('passwordHash');
    });

    it('用户名已存在时返回 409', async () => {
      const res = await request(app)
        .post('/api/v1/users')
        .set('Authorization', `Bearer ${authToken}`)
        .send({
          username: 'testuser',
          password: 'password123',
        });

      expect(res.status).toBe(409);
      expect(res.body.code).toBe(10002);
      expect(res.body.message).toContain('已存在');
    });

    it('缺少必填字段时返回 400', async () => {
      const res = await request(app)
        .post('/api/v1/users')
        .set('Authorization', `Bearer ${authToken}`)
        .send({
          username: 'newuser',
          // 缺少 password
        });

      expect(res.status).toBe(400);
      expect(res.body.code).toBe(40000);
      expect(res.body.errors).toBeInstanceOf(Array);
    });

    it('未授权访问返回 401', async () => {
      const res = await request(app)
        .post('/api/v1/users')
        .send({
          username: 'newuser2',
          password: 'password123',
        });

      expect(res.status).toBe(401);
      expect(res.body.code).toBe(10001);
    });
  });

  describe('GET /api/v1/users/:id', () => {
    it('应该成功获取用户信息', async () => {
      const res = await request(app)
        .get(`/api/v1/users/${testUser.id}`)
        .set('Authorization', `Bearer ${authToken}`);

      expect(res.status).toBe(200);
      expect(res.body.code).toBe(0);
      expect(res.body.data.username).toBe('testuser');
      expect(res.body.data).not.toHaveProperty('passwordHash');
    });

    it('用户不存在返回 404', async () => {
      const res = await request(app)
        .get('/api/v1/users/non-existent-id')
        .set('Authorization', `Bearer ${authToken}`);

      expect(res.status).toBe(404);
      expect(res.body.code).toBe(10003);
    });
  });
});

### 8.3.2 E2E 测试示例

**测试数据清理说明：**

E2E 测试会产生大量测试数据，需要在测试前后进行清理，避免数据污染和影响后续测试。

```typescript
// e2e/utils/test-data-helper.ts
import { request } from '@playwright/test';

export class TestDataCleaner {
  private apiToken: string;
  private createdUsers: string[] = [];
  private createdOrders: string[] = [];

  constructor(apiToken: string) {
    this.apiToken = apiToken;
  }

  // 注册清理任务
  trackUser(userId: string) {
    this.createdUsers.push(userId);
  }

  trackOrder(orderId: string) {
    this.createdOrders.push(orderId);
  }

  // 清理所有测试数据
  async cleanup() {
    const apiContext = await request.newContext({
      baseURL: 'http://localhost:3000',
      extraHTTPHeaders: {
        Authorization: `Bearer ${this.apiToken}`,
      },
    });

    // 批量删除用户
    for (const userId of this.createdUsers) {
      try {
        await apiContext.delete(`/api/v1/users/${userId}`);
      } catch (e) {
        console.warn(`Failed to delete user ${userId}:`, e);
      }
    }

    // 批量删除订单
    for (const orderId of this.createdOrders) {
      try {
        await apiContext.delete(`/api/v1/orders/${orderId}`);
      } catch (e) {
        console.warn(`Failed to delete order ${orderId}:`, e);
      }
    }

    await apiContext.dispose();
  }
}

// 数据库直接清理（开发环境）
export async function cleanupDatabase(pool: Pool) {
  await pool.query('SET FOREIGN_KEY_CHECKS = 0');
  await pool.query('TRUNCATE TABLE test_users');
  await pool.query('TRUNCATE TABLE test_orders');
  await pool.query('TRUNCATE TABLE test_audit_logs');
  await pool.query('SET FOREIGN_KEY_CHECKS = 1');
}
```

// e2e/user-management.spec.ts
import { test, expect } from '@playwright/test';
import { TestDataCleaner } from './utils/test-data-helper';

test.describe('用户管理', () => {
  let cleaner: TestDataCleaner;
  let apiToken: string;

  test.beforeEach(async ({ page }) => {
    // 登录获取 token
    await page.goto('/login');
    await page.fill('input[name="username"]', 'admin');
    await page.fill('input[name="password"]', 'admin123');
    await page.click('button[type="submit"]');
    await page.waitForURL('/dashboard');

    // 从 localStorage 或 Cookie 获取 token
    const storage = await page.context().storageState();
    apiToken = storage.origins[0].cookies.find(c => c.name === 'token')?.value;
    cleaner = new TestDataCleaner(apiToken);
  });

  test.afterEach(async () => {
    // 清理测试数据
    await cleaner.cleanup();
  });

  test('应该可以创建新用户', async ({ page }) => {
    await page.goto('/users');

    await page.click('button:has-text("新建")');

    await page.fill('input[name="username"]', 'e2etest');
    await page.fill('input[name="password"]', 'password123');
    await page.fill('input[name="nickname"]', 'E2E 测试用户');

    await page.click('button:has-text("确定")');

    await expect(page.locator('.el-message--success'))

    await expect(page.locator('text=e2etest')).toBeVisible();

**文件命名格式：**

**规范说明：**

|------|------|------|------|

### 8.4.2 AAA 模式

**测试 Arrange-Act-Assert 详解：**

AAA 模式是单元测试的标准结构，将测试分为三个明确的阶段：

1. **Arrange（准备）**：设置测试所需的所有前置条件和数据
2. **Act（执行）**：执行被测试的操作
3. **Assert（断言）**：验证操作结果是否符合预期

```typescript
// ✅ 推荐：清晰的 AAA 结构
it('应该计算正确的总价', () => {
  // Arrange - 准备测试数据
  const cart = new ShoppingCart();
  cart.addItem({ price: 100, quantity: 2 });  // 200 元
  cart.addItem({ price: 50, quantity: 3 });   // 150 元

  // Act - 执行被测试的操作
  const total = cart.getTotal();

  // Assert - 验证结果
  expect(total).toBe(350);
});

// ✅ 推荐：复杂场景的 AAA 结构
it('应该应用折扣当订单金额满 1000 时', () => {
  // Arrange
  const discountRule = new DiscountRule({
    minAmount: 1000,
    discountRate: 0.9,  // 9 折
  });
  const order = new Order();
  order.addItem({ price: 500, quantity: 3 });  // 1500 元

  // Act
  const finalAmount = discountRule.apply(order);

  // Assert
  expect(finalAmount).toBe(1350);  // 1500 * 0.9
});

// ✅ 推荐：异常测试的 AAA 结构
it('应该抛出 InsufficientStockError 当库存不足时', () => {
  // Arrange
  const inventory = new Inventory();
  inventory.addProduct({ id: 'p1', stock: 5 });
  const orderItem = { productId: 'p1', quantity: 10 };

  // Act & Assert
  expect(() => inventory.fulfillOrder(orderItem))
    .toThrow(InsufficientStockError);
});
```

**AAA 模式最佳实践：**

```typescript
// 1. 每个阶段之间用空行分隔，提高可读性
it('应该有清晰的阶段分隔', () => {
  // Arrange
  const user = new User();

  // Act
  user.activate();

  // Assert
  expect(user.isActive).toBe(true);
});

// 2. Arrange 部分可以使用测试工厂简化
it('应该使用工厂函数简化准备', () => {
  // Arrange
  const user = createUser({ status: 'pending' });
  const role = createRole({ name: 'admin' });

  // Act
  user.assignRole(role);

  // Assert
  expect(user.roles).toContain(role);
});

// 3. Act 部分应该只调用一个方法
it('应该只执行一个操作', () => {
  // Arrange
  const calculator = new Calculator();
  calculator.set(10);

  // Act
  const result = calculator.add(5);

  // Assert
  expect(result).toBe(15);
});

// 4. Assert 部分可以有多个断言，但都针对同一操作结果
it('可以验证多个方面', () => {
  // Arrange
  const service = new UserService();
  const dto = { username: 'test', email: 'test@example.com' };

  // Act
  const user = service.create(dto);

  // Assert - 多个断言验证同一结果
  expect(user.username).toBe('test');
  expect(user.email).toBe('test@example.com');
  expect(user.id).toBeDefined();
  expect(user.createdAt).toBeInstanceOf(Date);
});
```

      detail: faker.location.streetAddress(),
    },
    remark: overrides.remark ?? faker.lorem.sentence(),
    createdAt: overrides.createdAt ?? new Date(),
    updatedAt: overrides.updatedAt ?? new Date(),

// services/order.service.spec.ts
import { OrderService } from './order.service';
import { createOrder, createPaidOrder } from '../../test/factories/order.factory';
import { createUser } from '../../test/factories/user.factory';

describe('OrderService', () => {
  let orderService: OrderService;
  let mockOrderRepository: any;

  beforeEach(() => {
    mockOrderRepository = {
      findById: jest.fn(),
      create: jest.fn(),
      updateStatus: jest.fn(),
    };
    orderService = new OrderService(mockOrderRepository);
  });

  describe('cancelOrder', () => {
    it('应该取消订单当订单状态为待支付时', async () => {
      // Arrange
      const order = createOrder({ status: OrderStatus.PENDING });
      mockOrderRepository.findById.mockResolvedValue(order);

      // Act
      await orderService.cancelOrder(order.id);

      // Assert
      expect(mockOrderRepository.updateStatus)
        .toHaveBeenCalledWith(order.id, OrderStatus.CANCELLED);
    });

    it('应该抛出错误当订单已支付时', async () => {
      // Arrange
      const order = createPaidOrder();
      mockOrderRepository.findById.mockResolvedValue(order);

      // Act & Assert
      await expect(orderService.cancelOrder(order.id))
        .rejects
        .toThrow('已支付的订单不能取消');
    });
  });
});
