### 8.1.2 单元测试结构

```typescript
// services/user.service.spec.ts
import { UserService } from './user.service';
import { UserDAO } from '../dao/user.dao';
import { RoleDAO } from '../dao/role.dao';
import { UserNotFoundError, UserExistsError } from '../errors';

describe('UserService', () => {
  let userService: UserService;
  let userDAO: Partial<UserDAO>;
  let roleDAO: Partial<RoleDAO>;

  beforeEach(() => {
    userDAO = {
      findById: jest.fn(),
      findByUsername: jest.fn(),
      create: jest.fn(),
      bindRole: jest.fn(),
    };
    roleDAO = {
      getDefault: jest.fn(),
    };
    userService = new UserService(userDAO as UserDAO, roleDAO as RoleDAO);
  });

  describe('getUserById', () => {
    it('应该成功获取用户', async () => {
      const mockUser = { id: '123', username: 'test' };
      (userDAO.findById as jest.Mock).mockResolvedValue(mockUser);

      const result = await userService.getUserById('123');

      expect(result).toEqual(mockUser);
      expect(userDAO.findById).toHaveBeenCalledWith('123');
    });

    it('用户不存在时应该抛出 UserNotFoundError', async () => {
      (userDAO.findById as jest.Mock).mockResolvedValue(null);

      await expect(userService.getUserById('999'))
        .rejects
        .toThrow(UserNotFoundError);
    });
  });

  describe('createUser', () => {
    it('应该成功创建用户', async () => {
      const dto = { username: 'newuser', password: 'password123' };
      const mockUser = { id: '123', ...dto };
      const mockRole = { id: 'role-1' };

      (userDAO.findByUsername as jest.Mock).mockResolvedValue(null);
      (userDAO.create as jest.Mock).mockResolvedValue(mockUser);
      (roleDAO.getDefault as jest.Mock).mockResolvedValue(mockRole);

      const result = await userService.createUser(dto);

      expect(result.username).toBe('newuser');
      expect(userDAO.create).toHaveBeenCalled();
      expect(userDAO.bindRole).toHaveBeenCalledWith('123', 'role-1');
    });

    it('用户名已存在时应该抛出 UserExistsError', async () => {
      const dto = { username: 'existing', password: 'password123' };
      (userDAO.findByUsername as jest.Mock).mockResolvedValue({ id: '1' });

      await expect(userService.createUser(dto))
        .rejects
        .toThrow(UserExistsError);
    });
  });
});
```

### 8.2.2 集成测试示例

```typescript
// test/integration/user-api.spec.ts
import request from 'supertest';
import { app } from '../../src/main';
import { testConnection } from '../database';

describe('User API Integration', () => {
  let authToken: string;

  beforeAll(async () => {
    // 创建测试用户并登录
    await request(app)
      .post('/api/v1/auth/register')
      .send({
        username: 'testuser',
        password: 'password123',
      });

    const res = await request(app)
      .post('/api/v1/auth/login')
      .send({
        username: 'testuser',
        password: 'password123',
      });

    authToken = res.body.data.accessToken;
  });

  describe('POST /api/v1/users', () => {
    it('应该成功创建用户', async () => {
      const res = await request(app)
        .post('/api/v1/users')
        .set('Authorization', `Bearer ${authToken}`)
        .send({
          username: 'newuser',
          password: 'password123',
          nickname: '测试用户',
        });

      expect(res.status).toBe(201);
      expect(res.body.code).toBe(0);
      expect(res.body.data.username).toBe('newuser');
    });

    it('用户名已存在时返回 409', async () => {
      const res = await request(app)
        .post('/api/v1/users')
        .set('Authorization', `Bearer ${authToken}`)
        .send({
          username: 'testuser',
          password: 'password123',
        });

      expect(res.status).toBe(409);
      expect(res.body.code).toBe(10002);
    });

    it('缺少必填字段时返回 400', async () => {
      const res = await request(app)
        .post('/api/v1/users')
        .set('Authorization', `Bearer ${authToken}`)
        .send({
          username: 'newuser',
          // 缺少 password
        });

      expect(res.status).toBe(400);
      expect(res.body.code).toBe(40000);
    });
  });
});
```

### 8.3.2 E2E 测试示例

```typescript
// e2e/user-management.spec.ts
import { test, expect } from '@playwright/test';

test.describe('用户管理', () => {
  test.beforeEach(async ({ page }) => {
    // 登录
    await page.goto('/login');
    await page.fill('input[name="username"]', 'admin');
    await page.fill('input[name="password"]', 'admin123');
    await page.click('button[type="submit"]');
    await page.waitForURL('/dashboard');
  });

  test('应该可以创建新用户', async ({ page }) => {
    // 导航到用户管理页面
    await page.goto('/users');

    // 点击新建按钮
    await page.click('button:has-text("新建")');

    // 填写表单
    await page.fill('input[name="username"]', 'e2etest');
    await page.fill('input[name="password"]', 'password123');
    await page.fill('input[name="nickname"]', 'E2E 测试用户');

    // 提交
    await page.click('button:has-text("确定")');

    // 验证成功提示
    await expect(page.locator('.el-message--success'))
      .toContainText('创建成功');

    // 验证列表中出现了新用户
    await expect(page.locator('text=e2etest')).toBeVisible();
  });

  test('应该可以删除用户', async ({ page }) => {
    await page.goto('/users');

    // 找到测试用户并删除
    const row = page.locator('tr', { hasText: 'e2etest' });
    await row.locator('button:has-text("删除")').click();

    // 确认删除
    await page.click('.el-message-box__footer button:has-text("确定")');

    // 验证成功提示
    await expect(page.locator('.el-message--success'))
      .toContainText('删除成功');

    // 验证列表中用户已消失
    await expect(page.locator('text=e2etest')).not.toBeVisible();
  });
});
```
