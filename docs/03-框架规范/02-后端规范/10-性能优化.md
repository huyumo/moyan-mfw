| 层级 | 说明 | 适用场景 | 示例 |
|------|------|----------|------|

### 10.1.3 缓存装饰器

```typescript
// decorators/cache.decorator.ts
import { SetMetadata, ExecutionContext, CallHandler } from '@nestjs/common';
import { Observable, of } from 'rxjs';
import { tap } from 'rxjs/operators';

export const CACHE_TTL_KEY = 'cache_ttl';
export const Cache = (ttlSeconds: number) => SetMetadata(CACHE_TTL_KEY, ttlSeconds);

@Injectable()
export class CacheInterceptor implements NestInterceptor {
  constructor(
    private redisService: RedisService,
    private reflector: Reflector,
  ) {}

  async intercept(
    context: ExecutionContext,
    next: CallHandler,
  ): Promise<Observable<any>> {
    const ttl = this.reflector.get(CACHE_TTL_KEY, context.getHandler());

    if (!ttl) {
      return next.handle();
    }

    const request = context.switchToHttp().getRequest();
    const cacheKey = this.generateCacheKey(request);

    // 尝试从缓存获取
    const cached = await this.redisService.get(cacheKey);
    if (cached) {
      return of(JSON.parse(cached));
    }

    // 执行请求并缓存结果
    return next.handle().pipe(
      tap(async (response) => {
        await this.redisService.setex(cacheKey, ttl, JSON.stringify(response));
      }),
    );
  }

  private generateCacheKey(request: Request): string {
    const { method, url } = request;
    return `${method}:${url}`;
  }
}
```

### 10.1.4 缓存更新策略

```typescript
// 缓存更新模式
export enum CacheStrategy {
  CACHE_ASIDE = 'CACHE_ASIDE',       // 旁路缓存：先读缓存，未命中则读库后写缓存
  WRITE_THROUGH = 'WRITE_THROUGH',   // 写透：同时写缓存和数据库
  WRITE_BEHIND = 'WRITE_BEHIND',     // 写回：先写缓存，异步写数据库
  INVALIDATION = 'INVALIDATION',     // 失效：数据更新时直接删除缓存
}

// 使用示例
@Injectable()
export class UserService {
  // 旁路缓存
  async getUserById(id: string): Promise<User> {
    const cacheKey = `user:${id}`;

    // 1. 读缓存
    const cached = await this.redis.get(cacheKey);
    if (cached) {
      return JSON.parse(cached);
    }

    // 2. 读数据库
    const user = await this.userDAO.findById(id);
    if (!user) {
      throw new UserNotFoundError();
    }

    // 3. 写缓存（300 秒过期）
    await this.redis.setex(cacheKey, 300, JSON.stringify(user));

    return user;
  }

  // 失效模式
  async updateUser(id: string, dto: UpdateUserDTO): Promise<User> {
    const cacheKey = `user:${id}`;

    // 1. 更新数据库
    const user = await this.userDAO.update(id, dto);

    // 2. 删除缓存
    await this.redis.del(cacheKey);

    return user;
  }
}
```

### 10.1.5 缓存击穿防护

```typescript
// 使用互斥锁防止缓存击穿
import { Mutex } from 'async-mutex';

const mutexes: Map<string, Mutex> = new Map();

async function getWithMutex<T>(
  key: string,
  fetchFn: () => Promise<T>,
  ttlSeconds: number
): Promise<T> {
  const cached = await redis.get(key);
  if (cached) {
    return JSON.parse(cached);
  }

  // 获取互斥锁
  let mutex = mutexes.get(key);
  if (!mutex) {
    mutex = new Mutex();
    mutexes.set(key, mutex);
  }

  const release = await mutex.acquire();
  try {
    // 双重检查
    const doubleCheck = await redis.get(key);
    if (doubleCheck) {
      return JSON.parse(doubleCheck);
    }

    // 从数据库加载
    const value = await fetchFn();
    await redis.setex(key, ttlSeconds, JSON.stringify(value));
    return value;
  } finally {
    release();
  }
}
```

### 10.1.6 热点数据防护

```typescript
// 热点 Key 永不过期 + 异步更新
export class HotKeyCache {
  private hotKeys: Set<string> = new Set();

  async get<T>(key: string): Promise<T | null> {
    const value = await redis.get(key);

    if (!value && this.hotKeys.has(key)) {
      // 热点 Key 失效，触发异步加载
      this.asyncRefresh(key);
      return null;  // 返回旧值或 null
    }

    return value ? JSON.parse(value) : null;
  }

  private async asyncRefresh(key: string): Promise<void> {
    // 异步刷新，不影响当前请求
    setTimeout(async () => {
      const value = await this.fetchFromDB(key);
      await redis.setex(key, 3600, JSON.stringify(value));
    }, 0);
  }
}
```

### 10.2.2 索引使用原则

```typescript
// ✅ 推荐：使用索引字段查询
const users = await userRepository.find({
  where: { username: 'test' },  // username 有索引

// ✅ 推荐：使用联合索引（符合最左前缀原则）

  where: {
    status: 1,         // 联合索引 (status, created_at) 第一列
    createdAt: date,   // 第二列
  },

// ❌ 避免：索引失效的查询
const users = await userRepository.find({
  where: {
    username: Like('%test%'),  // 左模糊查询，索引失效
  },
});

const users = await userRepository.find({
  where: {
    // 在索引列上使用函数，索引失效
    createdAt: MoreThan(new Date('2024-01-01')),
  },
});
```

### 10.2.3 分页优化

```typescript
// ✅ 推荐：游标分页（大数据量场景）
async function getCursorPagination(
  cursor: string,
  limit: number
): Promise<User[]> {
  return userRepository
    .createQueryBuilder('user')
    .where('user.id < :cursor', { cursor })
    .orderBy('user.id', 'DESC')
    .take(limit + 1)
    .getMany();
}

// ✅ 推荐：偏移分页 + 子查询优化
async function getOffsetPagination(
  page: number,
  pageSize: number
): Promise<User[]> {
  const offset = (page - 1) * pageSize;

  // 先获取 ID（覆盖索引）
  const idResult = await userRepository
    .createQueryBuilder('user')
    .select('user.id')
    .orderBy('user.id', 'DESC')
    .skip(offset)
    .take(pageSize)
    .getRawMany();

  const ids = idResult.map(r => r.id);

  // 通过 ID 查询完整数据
  return userRepository.findByIds(ids);
}

// ❌ 避免：深度分页
async function getDeepOffset(page: number, pageSize: number): Promise<User[]> {
  return userRepository.find({
    skip: (page - 1) * pageSize,  // 深度分页性能差
    take: pageSize,
  });
}
```

```typescript
// 限制并发数量
import pLimit from 'p-limit';

const limit = pLimit(5);  // 最多 5 个并发

async function processBatch(items: string[]) {
  const promises = items.map(item =>
    limit(async () => {
      await processItem(item);
    })
  );

  await Promise.all(promises);
}

// 批量处理大数组
async function processInBatches<T>(
  items: T[],
  batchSize: number,
  processor: (item: T) => Promise<void>
): Promise<void> {
  for (let i = 0; i < items.length; i += batchSize) {
    const batch = items.slice(i, i + batchSize);
    await Promise.all(batch.map(processor));
  }
}
```

### 10.3.3 竞态条件处理

```typescript
// ✅ 推荐：使用数据库锁
@Injectable()
export class InventoryService {
  async deductStock(productId: string, quantity: number): Promise<void> {
    await this.dataSource.transaction(async (manager) => {
      // 使用 FOR UPDATE 锁定行
      const product = await manager
        .createQueryBuilder(Product, 'product')
        .where('product.id = :id', { id: productId })
        .setLock('pessimistic_write')
        .getOne();

      if (!product || product.stock < quantity) {
        throw new StockNotEnoughError();
      }

      product.stock -= quantity;
      await manager.save(product);
    });
  }
}

// ✅ 推荐：使用乐观锁
@Entity('products')
export class Product {
  @PrimaryColumn()
  id: string;

  @Column()
  stock: number;

  @VersionColumn()  // 乐观锁版本号
  version: number;
}

async function updateWithOptimisticLock(
  productId: string,
  quantity: number
): Promise<void> {
  const result = await repository
    .createQueryBuilder()
    .update(Product)
    .set({ stock: () => `stock - ${quantity}` })
    .where('id = :id AND version = :version', {
      id: productId,
      version: currentVersion,
    })
    .execute();

  if (result.affected === 0) {
    throw new OptimisticLockError();  // 版本号不匹配
  }
}
```

```typescript
// 幂等性 Token
@Injectable()
export class IdempotencyService {
  async acquireToken(idempotencyKey: string): Promise<boolean> {
    const key = `idempotency:${idempotencyKey}`;
    // SETNX - 只有不存在时才能设置成功
    const acquired = await this.redis.set(key, '1', {
      NX: true,
      EX: 3600,  // 1 小时过期
    });

    return !!acquired;
  }

  async releaseToken(idempotencyKey: string): Promise<void> {
    const key = `idempotency:${idempotencyKey}`;
    await this.redis.del(key);
  }
}
```

### 10.4.1 性能指标采集

```typescript
// 使用 Prometheus 格式暴露指标
@Controller('metrics')
export class MetricsController {
  private metrics = {
    requestCount: 0,
    requestDuration: [],
    activeConnections: 0,
    heapUsed: 0,
  };

  recordRequest(duration: number) {
    this.metrics.requestCount++;
    this.metrics.requestDuration.push(duration);
    // 保留最近 1000 个数据点
    if (this.metrics.requestDuration.length > 1000) {
      this.metrics.requestDuration.shift();
    }
  }

  @Get()
  async metrics(): Promise<string> {
    const p99 = this.calculatePercentile(99);

    return `
# HELP http_requests_total 总请求数
# TYPE http_requests_total counter
http_requests_total ${this.metrics.requestCount}

# HELP http_request_duration_seconds 请求耗时 P99
# TYPE http_request_duration_seconds gauge
http_request_duration_seconds ${p99 / 1000}

# HELP node_heap_used_bytes 堆内存使用
# TYPE node_heap_used_bytes gauge
node_heap_used_bytes ${this.metrics.heapUsed}
`;
  }

  private calculatePercentile(p: number): number {
    const sorted = [...this.metrics.requestDuration].sort((a, b) => a - b);
    const index = Math.floor((sorted.length * p) / 100);
    return sorted[index] || 0;
  }
}
```
