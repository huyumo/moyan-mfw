## 9.1 环境配置

### 9.1.1 环境分类

| 环境 | 用途 | 域名示例 | 更新方式 |

| 开发环境 (dev) | 开发人员日常开发 | dev.api.moyan.com | 本地启动 |
| 测试环境 (test) | 测试人员功能测试 | test.api.moyan.com | 自动部署 |
| 预发布环境 (staging) | 上线前验证 | staging.api.moyan.com | 手动触发 |
| 生产环境 (prod) | 线上服务 | api.moyan.com | 审批后发布 |

## 9.2 Docker 容器化

### 9.2.1 Dockerfile

```dockerfile
# 多阶段构建 - 构建阶段
FROM node:18-alpine AS builder

WORKDIR /app

# 复制 package 文件
COPY package*.json ./
COPY pnpm-lock.yaml ./

# 安装依赖
RUN npm install -g pnpm
RUN pnpm install --frozen-lockfile

# 复制源代码
COPY . .

# 构建应用
RUN pnpm build

# 生产阶段
FROM node:18-alpine AS production

WORKDIR /app

# 创建非 root 用户
RUN addgroup -g 1001 -S nodejs
RUN adduser -S nestjs -u 1001

# 从构建阶段复制文件
COPY --from=builder /app/node_modules ./node_modules
COPY --from=builder /app/dist ./dist
COPY --from=builder /app/package.json ./

# 设置权限
RUN chown -R nestjs:nodejs /app

USER nestjs

# 暴露端口
EXPOSE 8080

# 健康检查
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
  CMD wget --no-verbose --tries=1 --spider http://localhost:8080/health || exit 1

# 启动应用
CMD ["node", "dist/main.js"]
```

### 9.2.2 Docker Compose

```yaml
# docker-compose.yml
version: '3.8'

services:
  app:
    build:
      context: .
      dockerfile: Dockerfile
    ports:
      - "8080:8080"
    environment:
      - NODE_ENV=production
      - DB_HOST=db
      - REDIS_HOST=redis
    depends_on:
      db:
        condition: service_healthy
      redis:
        condition: service_healthy
    restart: unless-stopped
    networks:
      - moyan-network

  db:
    image: mysql:8.0
    environment:
      MYSQL_ROOT_PASSWORD: ${DB_PASSWORD}
      MYSQL_DATABASE: moyan_prod
      MYSQL_USER: moyan_user
      MYSQL_PASSWORD: ${DB_PASSWORD}
    volumes:
      - mysql-data:/var/lib/mysql
    ports:
      - "3306:3306"
    healthcheck:
      test: ["CMD", "mysqladmin", "ping", "-h", "localhost"]
      interval: 10s
      timeout: 5s
      retries: 5
    networks:
      - moyan-network

  redis:
    image: redis:7-alpine
    volumes:
      - redis-data:/data
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5
    networks:
      - moyan-network

volumes:
  mysql-data:
  redis-data:

networks:
  moyan-network:
    driver: bridge
```

### 9.2.3 Kubernetes 部署

```yaml
# k8s/deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: moyan-backend
  namespace: production
spec:
  replicas: 3
  selector:
    matchLabels:
      app: moyan-backend
  template:
    metadata:
      labels:
        app: moyan-backend
    spec:
      containers:
        - name: backend
          image: moyan/backend:v1.0.0
          ports:
            - containerPort: 8080
          env:
            - name: NODE_ENV
              value: production
            - name: DB_HOST
              valueFrom:
                secretKeyRef:
                  name: db-credentials
                  key: host
          resources:
            requests:
              memory: "256Mi"
              cpu: "250m"
            limits:
              memory: "512Mi"
              cpu: "500m"
          livenessProbe:
            httpGet:
              path: /health/live
              port: 8080
            initialDelaySeconds: 30
            periodSeconds: 10
          readinessProbe:
            httpGet:
              path: /health/ready
              port: 8080
            initialDelaySeconds: 5
            periodSeconds: 5

---
apiVersion: v1
kind: Service
metadata:
  name: moyan-backend
spec:
  selector:
    app: moyan-backend
  ports:
    - port: 80
      targetPort: 8080
  type: ClusterIP
```

## 9.3 CI/CD 流程

### 9.3.1 GitHub Actions 配置

```yaml
# .github/workflows/deploy.yml
name: Deploy

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3

      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18'

      - name: Install pnpm
        run: npm install -g pnpm

      - name: Install dependencies
        run: pnpm install

      - name: Run lint
        run: pnpm lint

      - name: Run tests
        run: pnpm test:cov

      - name: Upload coverage
        uses: codecov/codecov-action@v3

  build:
    needs: test
    runs-on: ubuntu-latest
    if: github.event_name == 'push'
    steps:
      - uses: actions/checkout@v3

      - name: Build Docker image
        run: docker build -t moyan/backend:${{ github.sha }} .

      - name: Push to Registry
        run: |
          docker login -u ${{ secrets.DOCKER_USER }} -p ${{ secrets.DOCKER_PASSWORD }}
          docker push moyan/backend:${{ github.sha }}

  deploy-dev:
    needs: build
    runs-on: ubuntu-latest
    environment: development
    steps:
      - name: Deploy to Dev
        run: |
          kubectl set image deployment/moyan-backend \
            backend=moyan/backend:${{ github.sha }} \
            -n development

  deploy-prod:
    needs: build
    runs-on: ubuntu-latest
    environment: production
    steps:
      - name: Deploy to Production
        run: |
          kubectl set image deployment/moyan-backend \
            backend=moyan/backend:${{ github.sha }} \
            -n production
```

### 9.3.2 部署流程

```
代码提交 (git push)
       ↓
触发 CI 流水线
       ↓
代码检查 (Lint)
       ↓
单元测试 (Test)
       ↓
构建 Docker 镜像
       ↓
推送镜像到仓库
       ↓
自动部署到开发环境
       ↓
人工验证
       ↓
审批后部署到生产环境
```

### 9.3.3 回滚策略

```bash
# Kubernetes 回滚到上一个版本
kubectl rollout undo deployment/moyan-backend -n production

# 回滚到指定版本
kubectl rollout undo deployment/moyan-backend --to-revision=2 -n production

# 查看部署历史
kubectl rollout history deployment/moyan-backend -n production
```
