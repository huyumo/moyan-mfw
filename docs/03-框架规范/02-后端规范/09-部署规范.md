**🟢 推荐：HashiCorp Vault 集成**

```typescript
// config/vault.config.ts
import { Injectable, OnModuleInit } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { VaultClient } from 'hashicorp-vault';

@Injectable()
export class VaultConfigService implements OnModuleInit {
  private vaultClient: VaultClient;
  private secrets: Map<string, string> = new Map();

  constructor(private configService: ConfigService) {}

  async onModuleInit() {
    // 初始化 Vault 客户端
    this.vaultClient = new VaultClient({
      endpoint: this.configService.get('VAULT_ENDPOINT'),
      token: this.configService.get('VAULT_TOKEN'),
    });

    // 预加载密钥
    await this.loadSecrets();
  }

  /**
   * 从 Vault 加载密钥
   */
  private async loadSecrets(): Promise<void> {
    const secretPaths = [
      'secret/data/moyan/database',
      'secret/data/moyan/jwt',
      'secret/data/moyan/redis',
    ];

    for (const path of secretPaths) {
      try {
        const { data } = await this.vaultClient.read(path);
        for (const [key, value] of Object.entries(data.data)) {
          this.secrets.set(`${path}/${key}`, value as string);
        }
      } catch (error) {
        console.error(`Failed to load secret from ${path}:`, error);
      }
    }
  }

  /**
   * 获取密钥
   */
  getSecret(key: string): string | undefined {
    return this.secrets.get(key);
  }

  /**
   * 动态刷新密钥（定时任务）
   */
  async refreshSecrets(): Promise<void> {
    this.secrets.clear();
    await this.loadSecrets();
    console.log('Vault secrets refreshed');
  }
}

// main.ts
async function bootstrap() {
  const app = await NestFactory.create(AppModule);

  // 定时刷新密钥（每小时）
  const vaultService = app.get(VaultConfigService);
  setInterval(() => vaultService.refreshSecrets(), 60 * 60 * 1000);

  await app.listen(process.env.PORT || 3000);
}
bootstrap();
```

**🟢 推荐：AWS Secrets Manager 集成**

```typescript
// config/aws-secrets.config.ts
import {
  SecretsManagerClient,
  GetSecretValueCommand,
} from '@aws-sdk/client-secrets-manager';

@Injectable()
export class AwsSecretsService {
  private client: SecretsManagerClient;
  private cache: Map<string, any> = new Map();
  private cacheExpiry: Map<string, number> = new Map();

  constructor() {
    this.client = new SecretsManagerClient({
      region: process.env.AWS_REGION || 'ap-northeast-1',
    });
  }

  /**
   * 获取密钥（带缓存）
   */
  async getSecret(secretName: string): Promise<any> {
    const cached = this.cache.get(secretName);
    const expiry = this.cacheExpiry.get(secretName);

    // 缓存未过期
    if (cached && expiry && Date.now() < expiry) {
      return cached;
    }

    // 从 AWS Secrets Manager 获取
    const command = new GetSecretValueCommand({ SecretId: secretName });
    const response = await this.client.send(command);
    const secret = JSON.parse(response.SecretString!);

    // 缓存 5 分钟
    this.cache.set(secretName, secret);
    this.cacheExpiry.set(secretName, Date.now() + 5 * 60 * 1000);

    return secret;
  }
}

// 使用示例
const secrets = await awsSecretsService.getSecret('moyan/prod/database');
const dbPassword = secrets.password;
```

**🔴 强制：密钥管理最佳实践**

| 实践 | 说明 | 实现方式 |
|------|------|----------|
| 密钥不落地 | 不存储在代码库或配置文件 | Vault/AWS Secrets Manager |
| 密钥轮换 | 定期更换密钥 | Vault 自动轮换（30 天） |
| 最小权限 | 应用仅访问必需的密钥 | Vault Policy 限制 |
| 审计日志 | 记录密钥访问日志 | Vault Audit Device |
| 加密传输 | 密钥传输使用 TLS | HTTPS + 证书验证 |

**🟢 推荐：配置热重载**

```typescript
// config/config-hot-reload.service.ts
import { Injectable, OnModuleDestroy } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import chokidar from 'chokidar';

@Injectable()
export class ConfigHotReloadService implements OnModuleDestroy {
  private watcher: chokidar.FSWatcher;

  constructor(private configService: ConfigService) {
    this.setupWatcher();
  }

  /**
   * 监听配置文件变化
   */
  private setupWatcher() {
    const envFile = `.env.${process.env.NODE_ENV}`;

    this.watcher = chokidar.watch(envFile, {
      ignored: /node_modules/,
      persistent: true,
    });

    this.watcher.on('change', () => {
      console.log('配置文件变化，重新加载...');
      this.reloadConfig();
    });
  }

  /**
   * 重新加载配置
   */
  private async reloadConfig() {
    // 清除 require 缓存
    delete require.cache[require.resolve(`../${envFile}`)];

    // 重新加载环境变量
    const newEnv = require(`../${envFile}`);
    for (const [key, value] of Object.entries(newEnv)) {
      process.env[key] = value as string;
    }

    console.log('配置已重新加载');
  }

  onModuleDestroy() {
    this.watcher?.close();
  }
}
```

**配置分层加载顺序：**

```
1. process.env (系统环境变量)
       ↓
2. .env.[NODE_ENV] (环境特定配置)
       ↓
3. .env (通用配置)
       ↓
4. Vault / AWS Secrets Manager (密钥)
       ↓
5. 远程配置中心 (可选，如 Nacos/Apollo)
```

### 9.2.1 Dockerfile

**🟢 推荐：多阶段构建优化说明（镜像大小优化）**

```dockerfile
# ================================
# 阶段 1：依赖安装
# ================================
FROM node:18-alpine AS deps

RUN apk add --no-cache libc6-compat

WORKDIR /app

# 复制包管理文件
COPY package.json pnpm-lock.yaml ./

# 安装 pnpm 并安装依赖
RUN npm install -g pnpm && \
    pnpm install --frozen-lockfile --prefer-offline

# ================================
# 阶段 2：构建应用
# ================================
FROM node:18-alpine AS builder

WORKDIR /app

# 从 deps 阶段复制 node_modules
COPY --from=deps /app/node_modules ./node_modules
COPY . .

# 设置构建环境变量
ENV NODE_ENV=production

# 构建应用
RUN pnpm build

# 移除开发依赖
RUN pnpm prune --prod

# ================================
# 阶段 3：生产运行
# ================================
FROM node:18-alpine AS production

# 安装必要的运行时依赖
RUN apk add --no-cache dumb-init

WORKDIR /app

# 创建非 root 用户
RUN addgroup -g 1001 -S nodejs && \
    adduser -S nestjs -u 1001

# 从构建阶段复制文件
COPY --from=builder /app/node_modules ./node_modules
COPY --from=builder /app/dist ./dist
COPY --from=builder /app/package.json ./

# 设置权限
RUN chown -R nestjs:nodejs /app

USER nestjs

# 使用 dumb-init 处理信号
ENTRYPOINT ["dumb-init", "--"]

# 暴露端口
EXPOSE 8080

# 健康检查
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
  CMD wget --no-verbose --tries=1 --spider http://localhost:8080/health || exit 1

# 启动应用
CMD ["node", "dist/main.js"]
```

on:
  push:
    branches: [main, develop]

      - 'v*'

    branches: [main]

# 取消运行中的工作流（同一 PR）
concurrency:

env:

      - name: Install pnpm
        run: npm install -g pnpm

      - name: Install dependencies
        run: pnpm install --frozen-lockfile

      - name: Run lint
        run: pnpm lint

      - name: Run type check
        run: pnpm type-check

  # ================================
  # 2. 单元测试
  # ================================
  test:
    runs-on: ubuntu-latest
    needs: lint
    services:
      mysql:
        image: mysql:8.0
        env:
          MYSQL_ROOT_PASSWORD: test_password
          MYSQL_DATABASE: moyan_test
        options: >-
          --health-cmd="mysqladmin ping"
          --health-interval=10s
          --health-timeout=5s
          --health-retries=5
      redis:
        image: redis:7-alpine
        options: >-
          --health-cmd="redis-cli ping"
          --health-interval=10s
          --health-timeout=5s
          --health-retries=5

    env:
      DB_HOST: localhost
      DB_PORT: 3306
      DB_USERNAME: root
      DB_PASSWORD: test_password
      DB_NAME: moyan_test
      REDIS_HOST: localhost
      REDIS_PORT: 6379

    steps:
      - uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: ${{ env.NODE_VERSION }}
          cache: 'pnpm'

      - name: Install pnpm
        run: npm install -g pnpm

      - name: Install dependencies
        run: pnpm install --frozen-lockfile

      - name: Run tests with coverage
        run: pnpm test:cov

      - name: Upload coverage to Codecov
        uses: codecov/codecov-action@v3
        with:
          file: ./coverage/lcov.info
          fail_ci_if_error: false

  # ================================
  # 3. 构建 Docker 镜像
  # ================================
  build:
    needs: test
    runs-on: ubuntu-latest
    if: github.event_name == 'push'
    outputs:
      image_tag: ${{ steps.meta.outputs.tags }}
      image_digest: ${{ steps.build.outputs.digest }}

    steps:
      - uses: actions/checkout@v4

      # 使用 Docker Buildx 进行多平台构建
      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      # 登录镜像仓库
      - name: Login to Docker Hub
        uses: docker/login-action@v3
        with:
          username: ${{ secrets.DOCKER_USERNAME }}
          password: ${{ secrets.DOCKER_PASSWORD }}

      # 生成镜像元数据
      - name: Extract metadata
        id: meta
        uses: docker/metadata-action@v5
        with:
          images: ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}
          tags: |
            type=ref,event=branch
            type=ref,event=pr
            type=semver,pattern={{version}}
            type=semver,pattern={{major}}.{{minor}}
            type=sha,prefix=sha-
            type=raw,value=latest,enable={{is_default_branch}}

      # 构建并推送
      - name: Build and push
        id: build
        uses: docker/build-push-action@v5
        with:
          context: .
          push: true
          tags: ${{ steps.meta.outputs.tags }}
          labels: ${{ steps.meta.outputs.labels }}
          cache-from: type=gha
          cache-to: type=gha,mode=max
          platforms: linux/amd64,linux/arm64

      # 安全扫描
      - name: Run Trivy vulnerability scanner
        uses: aquasecurity/trivy-action@master
        with:
          image-ref: ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:${{ github.sha }}
          format: 'sarif'
          output: 'trivy-results.sarif'
          severity: 'CRITICAL,HIGH'

      - name: Upload Trivy scan results
        uses: github/codeql-action/upload-sarif@v2
        with:
          sarif_file: 'trivy-results.sarif'

  # ================================
  # 4. 部署到开发环境
  # ================================
  deploy-dev:
    needs: build
    runs-on: ubuntu-latest
    environment:
      name: development
      url: https://dev.api.moyan.com
    steps:
      - uses: actions/checkout@v4

      - name: Setup kubectl
        uses: azure/setup-kubectl@v3
        with:
          version: 'v1.28.0'

      - name: Configure kubectl
        run: |
          echo "${{ secrets.KUBE_CONFIG_DEV }}" | base64 -d > kubeconfig
          export KUBECONFIG=kubeconfig

      - name: Deploy to Dev
        run: |
          kubectl set image deployment/moyan-backend \
            backend=${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:${{ github.sha }} \
            -n development

      - name: Wait for rollout
        run: |
          kubectl rollout status deployment/moyan-backend -n development --timeout=300s

      - name: Smoke test
        run: |
          curl -f https://dev.api.moyan.com/health || exit 1

  # ================================
  # 5. 部署到生产环境（需要审批）
  # ================================
  deploy-prod:
    needs: [build, deploy-dev]
    runs-on: ubuntu-latest
    environment:
      name: production
      url: https://api.moyan.com
    if: startsWith(github.ref, 'refs/tags/v')

    steps:
      - uses: actions/checkout@v4

      - name: Setup kubectl
        uses: azure/setup-kubectl@v3
        with:
          version: 'v1.28.0'

      - name: Configure kubectl
        run: |
          echo "${{ secrets.KUBE_CONFIG_PROD }}" | base64 -d > kubeconfig
          export KUBECONFIG=kubeconfig

      - name: Deploy to Production
        run: |
          kubectl set image deployment/moyan-backend \
            backend=${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:${{ github.ref_name }} \
            -n production

      - name: Wait for rollout
        run: |
          kubectl rollout status deployment/moyan-backend -n production --timeout=600s

      - name: Health check
        run: |
          curl -f https://api.moyan.com/health || exit 1

      - name: Notify success
        if: success()
        run: |

### 9.3.3 回滚策略

**🔴 强制：K8s 回滚命令详解**

```bash
# ================================
# 1. 快速回滚（到上一个稳定版本）
# ================================
# 最快速的回滚方式，Kubernetes 会自动保留旧 ReplicaSet
kubectl rollout undo deployment/moyan-backend -n production

# 等待回滚完成
kubectl rollout status deployment/moyan-backend -n production --timeout=300s

# ================================
# 2. 回滚到指定版本
# ================================
# 查看部署历史
kubectl rollout history deployment/moyan-backend -n production

# 输出示例：
# deployment/moyan-backend
# REVISION  CHANGE-CAUSE
# 1         Initial deployment
# 2         Update to v1.1.0
# 3         Update to v1.2.0
# 4         Update to v1.3.0 (当前版本，有问题)

# 回滚到 revision 3 (v1.2.0)
kubectl rollout undo deployment/moyan-backend --to-revision=3 -n production

# ================================
# 3. 暂停和恢复回滚
# ================================
# 暂停当前滚动更新（用于紧急情况）
kubectl rollout pause deployment/moyan-backend -n production

# 恢复滚动更新
kubectl rollout resume deployment/moyan-backend -n production

# ================================
# 4. 强制回滚（删除 Pod 重建）
# ================================
# 如果标准回滚失败，强制删除 Pod
kubectl delete pods -l app=moyan-backend -n production

# ================================
# 5. 使用 Helm 回滚
# ================================
# 查看 Helm release 历史
helm history moyan-backend -n production

# 回滚到上一个版本
helm rollback moyan-backend -n production

# 回滚到指定版本
helm rollback moyan-backend 3 -n production

# ================================
# 6. 使用 GitOps 回滚（ArgoCD）
# ================================
# 回滚 Git 仓库到上一个提交
git revert HEAD
git push origin main

# ArgoCD 会自动同步到集群
argocd app sync moyan-backend -n production
```

**🟢 推荐：回滚决策树**

```
发现生产问题
       ↓
评估问题严重程度
       ↓
┌──────────────────────────────────────────────────────┐
│  P0 严重（服务不可用/数据错误）                       │
│  → 立即回滚（无需审批）                              │
│  → 执行快速回滚命令                                  │
│  → 通知相关人员                                       │
└──────────────────────────────────────────────────────┘
       ↓
┌──────────────────────────────────────────────────────┐
│  P1 重要（部分功能异常/性能下降）                     │
│  → 技术负责人审批                                    │
│  → 尝试修复 or 回滚                                  │
│  → 如回滚，执行标准回滚流程                          │
└──────────────────────────────────────────────────────┘
       ↓
┌──────────────────────────────────────────────────────┐
│  P2 一般（轻微问题/不影响核心功能）                   │
│  → 产品负责人审批                                    │
│  → 优先修复，不回滚                                  │
└──────────────────────────────────────────────────────┘
```

**🟢 推荐：自动化回滚脚本**

```bash
#!/bin/bash
# scripts/auto-rollback.sh
# 自动回滚脚本：当健康检查失败时自动触发

set -e

NAMESPACE="production"
DEPLOYMENT="moyan-backend"
MAX_RETRIES=3
RETRY_COUNT=0

echo "🔍 检查部署健康状态..."

while [ $RETRY_COUNT -lt $MAX_RETRIES ]; do
  # 检查 Pod 状态
  READY_PODS=$(kubectl get pods -l app=$DEPLOYMENT -n $NAMESPACE \
    -o jsonpath='{.items[*].status.containerStatuses[0].ready}' | tr ' ' '\n' | grep -c true || true)

  TOTAL_PODS=$(kubectl get pods -l app=$DEPLOYMENT -n $NAMESPACE | wc -l)

  echo "就绪 Pod: $READY_PODS / $TOTAL_PODS"

  if [ "$READY_PODS" -eq "$TOTAL_PODS" ]; then
    # 检查健康端点
    HEALTH_RESPONSE=$(curl -s -o /dev/null -w "%{http_code}" https://api.moyan.com/health)

    if [ "$HEALTH_RESPONSE" == "200" ]; then
      echo "✅ 部署健康检查通过"
      exit 0
    fi
  fi

  RETRY_COUNT=$((RETRY_COUNT + 1))
  echo "⚠️  健康检查失败，等待重试 ($RETRY_COUNT/$MAX_RETRIES)..."
  sleep 30
done

echo "❌ 健康检查持续失败，触发自动回滚"

# 执行回滚
kubectl rollout undo deployment/$DEPLOYMENT -n $NAMESPACE

# 等待回滚完成
kubectl rollout status deployment/$DEPLOYMENT -n $NAMESPACE --timeout=300s

# 发送告警通知
curl -X POST "$DINGTALK_WEBHOOK" \
  -H "Content-Type: application/json" \
  -d "{
    \"msgtype\": \"text\",
    \"text\": {
      \"content\": \"🚨 自动回滚已触发\\n部署：$DEPLOYMENT\\n原因：健康检查失败\\n时间：$(date)\"
    }
  }"

echo "🔄 回滚完成"
```

**回滚策略对比：**

| 策略 | 适用场景 | 优点 | 缺点 |
|------|----------|------|------|
| 快速回滚 | P0 紧急故障 | 秒级恢复 | 可能丢失部分数据 |
| 滚动回滚 | 一般问题 | 零宕机 | 速度较慢 |
| 蓝绿切换 | 重大故障 | 快速且完整 | 资源成本高 |
| 金丝雀回滚 | 局部问题 | 影响范围小 | 配置复杂 |
