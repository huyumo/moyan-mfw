```typescript
// 使用 Redis 实现分布式限流
import { Injectable } from '@nestjs/common';
import { RedisService } from '../redis/redis.service';

@Injectable()
export class ThrottlerService {
  constructor(private redis: RedisService) {}

  async isRateLimited(key: string, limit: number, windowMs: number): Promise<boolean> {
    const now = Date.now();
    const windowKey = `throttle:${key}:${Math.floor(now / windowMs)}`;

    const count = await this.redis.incr(windowKey);
    if (count === 1) {
      await this.redis.expire(windowKey, Math.ceil(windowMs / 1000));
    }

    return count > limit;
  }
}

// 使用示例
const isLimited = await throttlerService.isRateLimited(
  `login:${req.ip}:${dto.username}`,
  5,      // 5 次
  900000  // 15 分钟
);

if (isLimited) {
  throw new TooManyRequestsError();
}
```

### 5.1.2 Token 结构

```typescript
// Access Token Payload
{
  "sub": "user-uuid",           // 用户 ID
  "username": "admin",          // 用户名
  "roles": ["admin", "user"],   // 角色列表
  "permissions": ["*:*:*"],     // 权限列表
  "iat": 1711771200,            // 签发时间
  "exp": 1711778400,            // 过期时间（2 小时）
  "iss": "moyan-mfw"            // 签发者
}

// Refresh Token Payload
{
  "sub": "user-uuid",
  "type": "refresh",            // Token 类型
  "iat": 1711771200,
  "exp": 1714363200,            // 过期时间（30 天）
  "jti": "unique-jti"           // 唯一标识，用于废除
}
```

**🔴 强制：Payload 字段安全说明**

| 字段 | 敏感性 | 说明 | 加密要求 |
|------|--------|------|----------|
| `sub` | 中 | 用户 ID | 建议使用 UUID，避免暴露真实 ID |
| `username` | 低 | 用户名 | 公开信息，无需加密 |
| `roles` | 中 | 角色列表 | 敏感角色建议用缩写 |
| `permissions` | 高 | 权限列表 | 数据量大时建议用哈希 |
| `iat`/`exp` | 低 | 时间戳 | 标准字段 |
| `iss` | 低 | 签发者 | 固定值 |
| `jti` | 高 | 唯一标识 | 用于废除 Token，必须唯一 |

**⚠️ 注意：Payload 不包含敏感信息**

```typescript
// ❌ 避免：Payload 中包含敏感信息
{
  "sub": "user-uuid",
  "password": "hashed-password",  // 不要包含密码哈希
  "phone": "13800138000",         // 不要包含个人隐私信息
  "email": "admin@example.com"    // 不要包含邮箱
}

// ✅ 推荐：Payload 仅包含认证授权必需信息
{
  "sub": "user-uuid",
  "username": "admin",
  "roles": ["admin"],
  "permissions": ["*:*:*"]
}
```

**🟢 推荐：Token 过期时间配置**

| Token 类型 | 推荐过期时间 | 使用场景 |
|------------|--------------|----------|
| Access Token | 2-4 小时 | 短期访问，减少泄露风险 |
| Refresh Token | 7-30 天 | 长期登录，支持无感刷新 |
| 记住我 Token | 90 天 | 用户主动选择"记住我" |

**Token 刷新机制：**

```typescript
// 双 Token 刷新流程
@Post('refresh')
async refreshToken(@Body('refreshToken') refreshToken: string) {
  // 1. 验证 Refresh Token
  const payload = await this.jwtService.verifyAsync(refreshToken, {
    secret: this.config.get('REFRESH_TOKEN_SECRET'),
  });

  // 2. 检查是否在黑名单中（被废除）
  const isBlacklisted = await this.redis.exists(`token:blacklist:${payload.jti}`);
  if (isBlacklisted) {
    throw new UnauthorizedError('Token 已废除');
  }

  // 3. 颁发新的 Access Token
  const newAccessToken = await this.jwtService.signAsync({
    sub: payload.sub,
    username: payload.username,
    roles: payload.roles,
  }, {
    secret: this.config.get('ACCESS_TOKEN_SECRET'),
    expiresIn: '2h',
  });

  return { accessToken: newAccessToken };
}
```

### 5.1.3 认证守卫实现

```typescript
// guards/auth.guard.ts
@Injectable()
export class AuthGuard implements CanActivate {
  constructor(
    private jwtService: JwtService,
    private reflector: Reflector,
  ) {}

  async canActivate(context: ExecutionContext): Promise<boolean> {
    // 检查是否跳过认证
    const isPublic = this.reflector.getAllAndOverride<boolean>(PUBLIC_KEY, [
      context.getHandler(),
      context.getClass(),
    ]);
    if (isPublic) return true;

    const request = context.switchToHttp().getRequest();
    const token = this.extractTokenFromHeader(request);

    if (!token) {
      throw new UnauthorizedError();
    }

    try {
      const payload = await this.jwtService.verifyAsync(token);
      request.user = payload;
    } catch (error) {
      throw new UnauthorizedError();
    }

    return true;
  }
}

// 使用
@Controller('users')
@UseGuards(AuthGuard)
export class UserController {
  // ...
}

// 跳过认证
@Controller('auth')
export class AuthController {
  @Post('login')
  @Public()  // 跳过认证
  async login(@Body() dto: LoginDTO) {
    // ...
  }
}
```

```typescript
// 从请求头提取 Token
private extractTokenFromHeader(request: Request): string | undefined {
  const [type, token] = request.headers.authorization?.split(' ') ?? [];
  return type === 'Bearer' ? token : undefined;
}

// 支持多种 Token 来源
private extractToken(request: Request): string | undefined {
  // 1. 从 Authorization 头
  const [type, token] = request.headers.authorization?.split(' ') ?? [];
  if (type === 'Bearer') return token;

  // 2. 从 Cookie
  const cookieToken = request.cookies?.['access_token'];
  if (cookieToken) return cookieToken;

  // 3. 从查询参数（不推荐，仅用于 WebSocket）
  const queryToken = request.query['token'];
  if (queryToken) return queryToken;

  return undefined;
}
```

```typescript
// 角色层级定义
export const ROLE_HIERARCHY = {
  'super_admin': ['admin', 'user', 'guest'],        // 超级管理员继承所有角色
  'admin': ['user', 'guest'],                        // 管理员继承普通用户
  'dept_manager': ['user', 'guest'],                 // 部门经理继承普通用户
  'user': ['guest'],                                 // 普通用户继承访客
  'guest': [],                                       // 访客无继承
};

// 角色继承守卫
@Injectable()
export class RolesGuard implements CanActivate {
  constructor(private reflector: Reflector) {}

  canActivate(context: ExecutionContext): boolean {
    const requiredRoles = this.reflector.getAllAndOverride<string[]>(ROLES_KEY, [
      context.getHandler(),
      context.getClass(),
    ]);

    if (!requiredRoles?.length) return true;

    const { user } = context.switchToHttp().getRequest();
    if (!user?.roles) return false;

    // 获取用户所有角色（包含继承角色）
    const allUserRoles = this.getAllInheritedRoles(user.roles);

    // 检查是否有任一要求的角色
    return requiredRoles.some((role) => allUserRoles.includes(role));
  }

  // 递归获取所有继承角色
  private getAllInheritedRoles(roles: string[]): string[] {
    const allRoles = new Set<string>(roles);

    for (const role of roles) {
      const inheritedRoles = ROLE_HIERARCHY[role] || [];
      const childRoles = this.getAllInheritedRoles(inheritedRoles);
      childRoles.forEach((r) => allRoles.add(r));
    }

    return Array.from(allRoles);
  }
}

// 使用示例
// 用户角色：dept_manager
// 继承角色：dept_manager → user → guest

@Controller('users')
@UseGuards(AuthGuard, RolesGuard)
@Roles('user')  // dept_manager 也可以访问（因为继承）
export class UserController {
  // ...
}
```

```typescript
      permCode: string;
```

**🔴 强制：权限位运算工具函数**

```typescript
// utils/permission-utils.ts
import { Perm } from './permission.enum';

/**
 * 权限工具类
 */
export class PermissionUtils {
  /**
   * 合并多个权限位
   * @example
   * mergePermissions(Perm.ADD, Perm.EDIT, Perm.DELETE) // returns 7n (0000111)
   */
  static mergePermissions(...perms: Perm[]): bigint {
    return perms.reduce((acc, perm) => acc | perm, 0n);
  }

  /**
   * 检查是否有指定权限
   * @example
   * hasPermission(7n, Perm.ADD)     // true
   * hasPermission(7n, Perm.EDIT)    // true
   * hasPermission(7n, Perm.EXPORT)  // false
   */
  static hasPermission(permValue: bigint, required: Perm): boolean {
    return (permValue & required) === required;
  }

  /**
   * 检查是否有任一权限
   * @example
   * hasAnyPermission(7n, [Perm.ADD, Perm.EXPORT])  // true (有 ADD 权限)
   */
  static hasAnyPermission(permValue: bigint, required: Perm[]): boolean {
    return required.some((perm) => this.hasPermission(permValue, perm));
  }

  /**
   * 检查是否有所有权限
   * @example
   * hasAllPermissions(7n, [Perm.ADD, Perm.EDIT])  // true
   * hasAllPermissions(7n, [Perm.ADD, Perm.EXPORT])  // false
   */
  static hasAllPermissions(permValue: bigint, required: Perm[]): boolean {
    return required.every((perm) => this.hasPermission(permValue, perm));
  }

  /**
   * 移除指定权限
   * @example
   * removePermission(7n, Perm.DELETE)  // returns 3n (0000011)
   */
  static removePermission(permValue: bigint, remove: Perm): bigint {
    return permValue & ~remove;
  }

  /**
   * 从权限码字符串解析权限位
   * @example
   * // 数据库存储：system.user:7  (permCode:permValue)
   * parsePermissionCode('system.user:7')  // { code: 'system.user', value: 7n }
   */
  static parsePermissionCode(codeWithBits: string): { code: string; value: bigint } {
    const [code, bitsStr] = codeWithBits.split(':');
    return {
      code,
      value: BigInt(bitsStr || '0'),
    };
  }

  /**
   * 生成权限码字符串（用于持久化）
   * @example
   * formatPermissionCode('system.user', 7n)  // 'system.user:7'
   */
  static formatPermissionCode(code: string, value: bigint): string {
    return `${code}:${value.toString()}`;
  }
}

// 权限位定义
export enum Perm {
  ADD     = 1n,    // 0000001 - 新增
  EDIT    = 2n,    // 0000010 - 编辑
  DELETE  = 4n,    // 0000100 - 删除
  EXPORT  = 8n,    // 0001000 - 导出
  IMPORT  = 16n,   // 0010000 - 导入
  VIEW    = 32n,   // 0100000 - 查看
  APPROVE = 64n,   // 1000000 - 审批
}
```

**使用示例：**

```typescript
// Service 层：获取用户权限值
@Injectable()
export class PermissionService {
  async getUserPermissions(userId: string): Promise<string[]> {
    // 从数据库获取用户的权限码列表
    // 存储格式：['system.user:7', 'system.role:3', ...]
    const permissionCodes = await this.permissionRepo.find({
      where: { userId },
      select: ['permCode', 'permValue'],
    });

    return permissionCodes.map(p => `${p.permCode}:${p.permValue}`);
  }

  // 检查权限
  async checkPermission(
    userId: string,
    permCode: string,
    requiredPerms: Perm[]
  ): Promise<boolean> {
    const permissions = await this.getUserPermissions(userId);

    // 查找匹配的权限码
    const permission = permissions.find(p => p.startsWith(`${permCode}:`));
    if (!permission) return false;

    const { value } = PermissionUtils.parsePermissionCode(permission);
    return PermissionUtils.hasAllPermissions(value, requiredPerms);
  }
}
```

```typescript
// 权限初始化服务（应用启动时加载权限定义）
@Injectable()
export class PermissionInitService {
  private readonly BUILTIN_PERMISSIONS = [
    { code: 'system.user', name: '用户管理', value: 7n },
    { code: 'system.role', name: '角色管理', value: 3n },
    { code: 'system.perm', name: '权限管理', value: 63n },
  ];

  async initialize(): Promise<void> {
    for (const perm of this.BUILTIN_PERMISSIONS) {
      await this.permissionRepo.upsert(
        {
          permCode: perm.code,
          permName: perm.name,
          permValue: perm.value,
          status: 1,
```

```typescript
  const parts = hash.split('$');
```

```typescript
import { createCipheriv, createDecipheriv } from 'crypto';

const ALGORITHM = 'aes-256-gcm';
const IV_LENGTH = 16;
const KEY = Buffer.from(process.env.ENCRYPTION_KEY!, 'hex');

// 加密
function encrypt(text: string): string {
  const iv = crypto.randomBytes(IV_LENGTH);
  const cipher = createCipheriv(ALGORITHM, KEY, iv);
  let encrypted = cipher.update(text, 'utf8', 'hex');
  encrypted += cipher.final('hex');
  const authTag = cipher.getAuthTag().toString('hex');
  return `${authTag}:${iv.toString('hex')}:${encrypted}`;
}

// 解密
function decrypt(encrypted: string): string {
  const [authTag, iv, encryptedText] = encrypted.split(':');
  const decipher = createDecipheriv(
    ALGORITHM,
    KEY,
    Buffer.from(iv, 'hex'),
  );
  decipher.setAuthTag(Buffer.from(authTag, 'hex'));
  let decrypted = decipher.update(encryptedText, 'hex', 'utf8');
  decrypted += decipher.final('utf8');
  return decrypted;
}

  }

  }
```

```typescript
// 证书过期检查
import https from 'https';
import { TLSSocket } from 'tls';

async function checkCertificateExpiry(domain: string): Promise<{ valid: boolean; daysRemaining: number }> {
  return new Promise((resolve, reject) => {
    const options = {
      hostname: domain,
      port: 443,
      method: 'HEAD',
    };

    const req = https.request(options, (res) => {
      const socket = res.socket as TLSSocket;
      const cert = socket.getPeerCertificate(true);

      if (!cert || !cert.valid_to) {
        resolve({ valid: false, daysRemaining: 0 });
        return;
      }

      const validTo = new Date(cert.valid_to);
      const now = new Date();
      const daysRemaining = Math.ceil((validTo.getTime() - now.getTime()) / (1000 * 60 * 60 * 24));

      resolve({ valid: daysRemaining > 30, daysRemaining });
    });

    req.on('error', (error) => {
      reject(error);
    });

    req.end();
  });
}

// 定时检查（每天）
async function dailyCertificateCheck() {
  const domains = ['example.com', 'api.example.com'];

  for (const domain of domains) {
    try {
      const result = await checkCertificateExpiry(domain);
      if (result.daysRemaining < 30) {
        // 发送告警通知
        await alertService.sendWarning(`证书即将过期：${domain}，剩余${result.daysRemaining}天`);
      }
    } catch (error) {
      await alertService.sendError(`证书检查失败：${domain}`);
    }
  }
}
```

### 5.4.3 速率限制

**🔴 强制：登录接口限流（已在 5.1.1 定义）**

详见 [5.1.1 JWT Token 认证](#5.1.1-jwt-token-认证) 中的登录限流要求。

**🟢 推荐：分布式限流实现（Redis）**

```typescript
// services/throttler.service.ts
import { Injectable, Inject } from '@nestjs/common';
import { RedisClientType } from 'redis';

@Injectable()
export class ThrottlerService {
  constructor(
    @Inject('REDIS_CLIENT') private redisClient: RedisClientType,
  ) {}

  /**
   * 滑动窗口限流算法
   * @param key 限流 Key（如：ip:username）
   * @param limit 限制次数
   * @param windowMs 时间窗口（毫秒）
   * @returns 是否被限流
   */
  async isRateLimited(
    key: string,
    limit: number,
    windowMs: number,
  ): Promise<{ limited: boolean; remaining: number; resetAt: number }> {
    const now = Date.now();
    const windowKey = `throttle:${key}:${Math.floor(now / windowMs)}`;

    // 原子操作：自增并设置过期时间
    const multi = this.redisClient.multi();
    multi.incr(windowKey);
    multi.expire(windowKey, Math.ceil(windowMs / 1000));
    const results = await multi.exec();

    const count = results[0] as number;
    const remaining = Math.max(0, limit - count);
    const resetAt = Math.ceil((Math.floor(now / windowMs) + 1) * windowMs);

    return {
      limited: count > limit,
      remaining,
      resetAt,
    };
  }

  /**
   * 固定窗口限流（简单场景）
   */
  async hitWithFixedWindow(
    key: string,
    limit: number,
    windowMs: number,
  ): Promise<boolean> {
    const windowKey = `throttle:fixed:${key}:${Math.floor(Date.now() / windowMs)}`;
    const count = await this.redisClient.incr(windowKey);

    if (count === 1) {
      await this.redisClient.expire(windowKey, Math.ceil(windowMs / 1000));
    }

    return count > limit;
  }

  /**
   * 令牌桶限流
   */
  async hitWithTokenBucket(
    key: string,
    capacity: number,
    refillRate: number, // 每秒补充的令牌数
  ): Promise<{ allowed: boolean; remaining: number }> {
    const bucketKey = `bucket:${key}`;
    const now = Date.now();

    // 使用 Lua 脚本保证原子性
    const luaScript = `
      local key = KEYS[1]
      local capacity = tonumber(ARGV[1])
      local refillRate = tonumber(ARGV[2])
      local now = tonumber(ARGV[3])

      local bucket = redis.call('HMGET', key, 'tokens', 'lastRefill')
      local tokens = tonumber(bucket[1]) or capacity
      local lastRefill = tonumber(bucket[2]) or now

      -- 计算应补充的令牌数
      local elapsed = (now - lastRefill) / 1000
      local refilledTokens = math.min(capacity, tokens + (elapsed * refillRate))

      if refilledTokens >= 1 then
        redis.call('HMSET', key, 'tokens', refilledTokens - 1, 'lastRefill', now)
        redis.call('EXPIRE', key, 3600)
        return {1, refilledTokens - 1}
      else
        return {0, refilledTokens}
      end
    `;

    const result = await this.redisClient.eval(luaScript, {
      keys: [bucketKey],
      arguments: [capacity.toString(), refillRate.toString(), now.toString()],
    });

    return {
      allowed: result[0] === 1,
      remaining: Math.floor(result[1]),
    };
  }
}

// 使用示例（Guard 方式）
@Injectable()
export class ThrottleGuard implements CanActivate {
  constructor(
    private throttlerService: ThrottlerService,
    private reflector: Reflector,
  ) {}

  async canActivate(context: ExecutionContext): Promise<boolean> {
    const request = context.switchToHttp().getRequest();
    const limit = this.reflector.get<number>('throttle:limit', context.getHandler()) || 10;
    const windowMs = this.reflector.get<number>('throttle:window', context.getHandler()) || 60000;

    const key = `api:${request.ip}:${request.route?.path || 'unknown'}`;
    const result = await this.throttlerService.isRateLimited(key, limit, windowMs);

    if (result.limited) {
      throw new TooManyRequestsException({
        code: 42900,
        message: '请求过于频繁，请稍后再试',
        retryAfter: Math.ceil((result.resetAt - Date.now()) / 1000),
      });
    }

    // 添加限流响应头
    const response = context.switchToHttp().getResponse();
    response.setHeader('X-RateLimit-Limit', limit);
    response.setHeader('X-RateLimit-Remaining', result.remaining);
    response.setHeader('X-RateLimit-Reset', result.resetAt);

    return true;
  }
}

// 装饰器
export const Throttle = (limit: number, windowMs: number) => {
  return SetMetadata('throttle:limit', limit) && SetMetadata('throttle:window', windowMs);
};

// Controller 中使用
@Controller('api')
@UseGuards(ThrottleGuard)
export class ApiController {
  @Post('search')
  @Throttle(20, 60000)  // 20 次/分钟
  async search(@Body() query: SearchDTO) {
    // ...
  }
}
```

```typescript
// 安全头检查中间件（开发/测试环境）
function securityHeadersCheck() {
  return (req: Request, res: Response, next: NextFunction) => {
    const requiredHeaders = {
      'Strict-Transport-Security': 'max-age=31536000',
      'X-Content-Type-Options': 'nosniff',
      'X-Frame-Options': 'DENY',
      'X-XSS-Protection': '1; mode=block',
      'Content-Security-Policy': undefined,  // 仅检查是否存在
    };

    for (const [header, expectedValue] of Object.entries(requiredHeaders)) {
      const actualValue = res.getHeader(header);
      if (!actualValue) {
        console.warn(`[Security] Missing header: ${header}`);
      } else if (expectedValue && !actualValue.includes(expectedValue)) {
        console.warn(`[Security] Header mismatch: ${header}`);
      }
    }

    next();
  };
}

// 仅在开发环境启用
if (process.env.NODE_ENV === 'development') {
  app.use(securityHeadersCheck());
}
```
