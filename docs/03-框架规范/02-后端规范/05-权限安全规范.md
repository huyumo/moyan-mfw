### 5.1.3 认证守卫实现

```typescript
// guards/auth.guard.ts
@Injectable()
export class AuthGuard implements CanActivate {
  constructor(
    private jwtService: JwtService,
    private reflector: Reflector,
  ) {}

  async canActivate(context: ExecutionContext): Promise<boolean> {
    // 检查是否跳过认证
    const isPublic = this.reflector.getAllAndOverride<boolean>(PUBLIC_KEY, [
      context.getHandler(),
      context.getClass(),
    ]);
    if (isPublic) return true;

    const request = context.switchToHttp().getRequest();
    const token = this.extractTokenFromHeader(request);

    if (!token) {
      throw new UnauthorizedError();
    }

    try {
      const payload = await this.jwtService.verifyAsync(token);
      request.user = payload;
    } catch (error) {
      throw new UnauthorizedError();
    }

    return true;
  }
}

// 使用
@Controller('users')
@UseGuards(AuthGuard)
export class UserController {
  // ...
}

// 跳过认证
@Controller('auth')
export class AuthController {
  @Post('login')
  @Public()  // 跳过认证
  async login(@Body() dto: LoginDTO) {
    // ...
  }
}
```

## 5.2 授权机制

### 5.2.1 基于角色的访问控制 (RBAC)

```typescript
// 角色守卫
@Injectable()
export class RolesGuard implements CanActivate {
  constructor(private reflector: Reflector) {}

  canActivate(context: ExecutionContext): boolean {
    const requiredRoles = this.reflector.getAllAndOverride<string[]>(ROLES_KEY, [
      context.getHandler(),
      context.getClass(),
    ]);

    if (!requiredRoles?.length) {
      return true;
    }

    const { user } = context.switchToHttp().getRequest();
    return requiredRoles.some((role) => user.roles?.includes(role));
  }
}

// 使用
@Controller('users')
@UseGuards(AuthGuard, RolesGuard)
@Roles('admin')  // 需要 admin 角色
export class UserController {
  @Delete(':id')
  async deleteUser(@Param('id') id: string) {
    // ...
  }
}
```

### 5.2.2 基于权限的访问控制

```typescript
// 权限装饰器
export const RequirePermission = (permCode: string, ...perms: Perm[]) =>
  SetMetadata('permissions', { permCode, perms });

// 权限守卫
@Injectable()
export class PermissionGuard implements CanActivate {
  async canActivate(context: ExecutionContext): Promise<boolean> {
    const { permCode, perms } = this.reflector.getAllAndOverride<{
      permCode: string;
      perms: Perm[];
    }>('permissions', [context.getHandler(), context.getClass()]);

    if (!permCode) return true;

    const { user } = context.switchToHttp().getRequest();
    const userPermissions = await this.permissionService.getUserPermissions(user.sub);

    // 检查是否有超级管理员权限
    if (userPermissions.includes('*:*:*')) return true;

    // 检查权限码和权限位
    return this.checkPermission(userPermissions, permCode, perms);
  }
}

// 使用
@Controller('users')
@UseGuards(AuthGuard, PermissionGuard)
export class UserController {
  @Delete(':id')
  @RequirePermission('system.user', Perm.DELETE)
  async deleteUser(@Param('id') id: string) {
    // ...
  }
}
```

### 5.2.3 权限位定义

```typescript
// 全局权限位枚举（使用 BigInt 位运算）
export enum Perm {
  ADD     = 1n,    // 0000001 - 新增
  EDIT    = 2n,    // 0000010 - 编辑
  DELETE  = 4n,    // 0000100 - 删除
  EXPORT  = 8n,    // 0001000 - 导出
  IMPORT  = 16n,   // 0010000 - 导入
  VIEW    = 32n,   // 0100000 - 查看
  APPROVE = 64n,   // 1000000 - 审批
}

// 权限值计算
const userPermValue = Perm.ADD | Perm.EDIT | Perm.DELETE;  // 0000111 = 7

// 权限检查
function hasPermission(permValue: bigint, required: Perm): boolean {
  return (permValue & required) === required;
}
```

## 5.3 敏感数据加密

### 5.3.1 密码加密

```typescript
import bcrypt from 'bcrypt';

const SALT_ROUNDS = 10;

// 加密密码
async function hashPassword(password: string): Promise<string> {
  return bcrypt.hash(password, SALT_ROUNDS);
}

// 验证密码
async function verifyPassword(
  password: string,
  hash: string
): Promise<boolean> {
  return bcrypt.compare(password, hash);
}

// 使用
@Injectable()
export class AuthService {
  async register(dto: RegisterDTO): Promise<User> {
    const hashedPassword = await hashPassword(dto.password);
    return this.userDAO.create({
      ...dto,
      passwordHash: hashedPassword,
    });
  }

  async login(dto: LoginDTO): Promise<{ accessToken: string }> {
    const user = await this.userDAO.findByUsername(dto.username);
    if (!user || !(await verifyPassword(dto.password, user.passwordHash))) {
      throw new InvalidCredentialsError();
    }
    // ...
  }
}
```

### 5.3.2 敏感字段加密存储

```typescript
import { createCipheriv, createDecipheriv } from 'crypto';

const ALGORITHM = 'aes-256-gcm';
const IV_LENGTH = 16;
const KEY = Buffer.from(process.env.ENCRYPTION_KEY!, 'hex');

// 加密
function encrypt(text: string): string {
  const iv = crypto.randomBytes(IV_LENGTH);
  const cipher = createCipheriv(ALGORITHM, KEY, iv);
  let encrypted = cipher.update(text, 'utf8', 'hex');
  encrypted += cipher.final('hex');
  const authTag = cipher.getAuthTag().toString('hex');
  return `${authTag}:${iv.toString('hex')}:${encrypted}`;
}

// 解密
function decrypt(encrypted: string): string {
  const [authTag, iv, encryptedText] = encrypted.split(':');
  const decipher = createDecipheriv(
    ALGORITHM,
    KEY,
    Buffer.from(iv, 'hex'),
  );
  decipher.setAuthTag(Buffer.from(authTag, 'hex'));
  let decrypted = decipher.update(encryptedText, 'hex', 'utf8');
  decrypted += decipher.final('utf8');
  return decrypted;
}

// 使用
@Entity('sys_secrets')
export class Secret {
  @Column({ name: 'encrypted_value', length: 500 })
  encryptedValue: string;  // 加密存储

  setValue(value: string) {
    this.encryptedValue = encrypt(value);
  }

  getValue(): string {
    return decrypt(this.encryptedValue);
  }
}
```

```typescript
function sanitizeInput(input: string): string {
  return sanitizeHtml(input, {
    allowedTags: [],  // 不允许任何标签
    allowedAttributes: {},
  });

// 使用
@Pipe()
```
