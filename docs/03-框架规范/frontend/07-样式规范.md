## 7.1 CSS 变量

### 7.1.1 主题变量

使用 CSS 变量定义主题颜色：

```scss
// 主题变量定义
:root {
  // 主色
  --mfw-color-primary: #1890ff;
  --mfw-color-primary-hover: #40a9ff;
  --mfw-color-primary-active: #096dd9;

  // 功能色
  --mfw-color-success: #52c41a;
  --mfw-color-warning: #faad14;
  --mfw-color-error: #f5222d;

  // 文本色
  --mfw-text-primary: #333333;
  --mfw-text-secondary: #666666;
  --mfw-text-disabled: #999999;

  // 背景色
  --mfw-bg-base: #ffffff;
  --mfw-bg-secondary: #f5f5f5;
}

// 暗黑模式
.mfw-theme-dark {
  --mfw-color-primary: #177ddc;
  --mfw-text-primary: #e8e8e8;
  --mfw-bg-base: #1f1f1f;
  --mfw-bg-secondary: #2b2b2b;
}
```

### 7.1.2 变量使用

```scss
.mfw-button {
  background-color: var(--mfw-color-primary);
  color: var(--mfw-text-primary);

  &:hover {
    background-color: var(--mfw-color-primary-hover);
  }
}
```

## 7.2 SCSS 编码规范

### 7.2.1 嵌套规则

```scss
// ✅ 推荐：合理嵌套（不超过 3 层）
.mfw-user-card {
  padding: 16px;

  .header {
    display: flex;

    .avatar {
      width: 48px;
      height: 48px;
    }
  }
}

// ❌ 避免：过度嵌套
.mfw-user-card {
  .header {
    .avatar {
      .img {
        .inner {
          // 嵌套过深
        }
      }
    }
  }
}
```

### 7.2.2 类名命名

```scss
// 使用 BEM 命名规范
.mfw-user-card { }                    // Block
.mfw-user-card__header { }            // Element
.mfw-user-card--compact { }           // Modifier

// 或使用短横线命名
.mfw-user-card { }
.mfw-user-card-header { }
.mfw-user-card-compact { }
```

### 7.2.3 Mixins

```scss
// 常用 Mixins
@mixin flex-center {
  display: flex;
  align-items: center;
  justify-content: center;
}

@mixin text-ellipsis {
  overflow: hidden;
  text-overflow: ellipsis;
  white-space: nowrap;
}

// 使用
.mfw-title {
  @include text-ellipsis;
}

.mfw-container {
  @include flex-center;
}
```

### 7.2.4 响应式

```scss
// 断点定义
$breakpoint-sm: 576px;
$breakpoint-md: 768px;
$breakpoint-lg: 992px;
$breakpoint-xl: 1200px;

// 响应式混入
@mixin respond-to($breakpoint) {
  @media (min-width: $breakpoint) {
    @content;
  }
}

// 使用
.mfw-container {
  width: 100%;

  @include respond-to($breakpoint-md) {
    width: 750px;
  }

  @include respond-to($breakpoint-lg) {
    width: 970px;
  }
}
```

## 7.3 scoped 样式

### 7.3.1 组件样式隔离

Vue 组件必须使用 `scoped`：

```vue
<style scoped lang="scss">
.mfw-user-card {
  padding: 16px;
}
</style>
```

### 7.3.2 深度选择器

需要修改子组件样式时使用 `:deep()`：

```vue
<style scoped lang="scss">
.mfw-user-card {
  :deep(.el-card__body) {
    padding: 24px;
  }
}
</style>
```

## 7.4 主题扩展

### 7.4.1 自定义主题

```typescript
// frontend/src/themes.ts
import type { ThemeRegistry } from 'moyan-mfw-base-frontend';

export const businessThemes: ThemeRegistry = {
  'business-light': {
    label: '商务浅色',
    tokens: {
      light: {
        primary: '#1890ff',
        success: '#52c41a',
        warning: '#faad14',
        error: '#f5222d',
      },
    },
  },
  'business-dark': {
    label: '商务深色',
    tokens: {
      dark: {
        primary: '#177ddc',
        success: '#389e0d',
        warning: '#d89614',
        error: '#a61d1d',
      },
    },
  },
};
```

### 7.4.2 主题注册

```typescript
// frontend/src/main.ts
import { createBaseAdminApp } from 'moyan-mfw-base-frontend';
import { businessThemes } from './themes';

const admin = createBaseAdminApp({
  title: '墨焱管理后台',
  themes: businessThemes,  // 注册自定义主题
  defaultTheme: 'business-light',
});
```
