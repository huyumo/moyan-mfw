## 3. 核心服务

### 3.1 权限计算服务

```typescript
// src/modules/permission/permission-calculation.service.ts
import { Injectable } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository, In } from 'typeorm';
import { Role } from '../entities/role.entity';
import { RolePermission } from '../entities/role-permission.entity';
import { User } from '../entities/user.entity';
import { Permission } from '../entities/permission.entity';

@Injectable()
export class PermissionCalculationService {
  constructor(
    @InjectRepository(Role)
    private roleRepository: Repository<Role>,
    @InjectRepository(RolePermission)
    private rolePermissionRepository: Repository<RolePermission>,
    @InjectRepository(Permission)
    private permissionRepository: Repository<Permission>,
  ) {}

  /**
   * 计算用户的最终权限
   * 用户最终权限 = 所有关联角色的 permissionValue 取 OR
   */
  async calculateUserPermissions(
    userId: string,
    appTypeId: string,
  ): Promise<Map<string, bigint>> {
    // 获取用户所有关联的角色
    const userRoles = await this.rolePermissionRepository
      .createQueryBuilder('ur')
      .innerJoin('ur.role', 'role')
      .where('ur.userId = :userId', { userId })
      .andWhere('role.appTypeId = :appTypeId', { appTypeId })
      .andWhere('role.roleStatus = 1', {})
      .getMany();

    // 按 permissionId 分组，合并权限值（位运算 OR）
    const permissionMap = new Map<string, bigint>();

    for (const rolePermission of userRoles) {
      const existingValue = permissionMap.get(rolePermission.permissionId) || 0n;
      permissionMap.set(rolePermission.permissionId, existingValue | rolePermission.permissionValue);
    }

    return permissionMap;
  }

  /**
   * 验证权限（子集检查）
   * (roleValue & requiredValue) === requiredValue
   */
  hasPermission(roleValue: bigint, requiredValue: bigint): boolean {
    return (roleValue & requiredValue) === requiredValue;
  }

  /**
   * 验证权限池约束
   * (poolValue & permValue) === permValue
   */
  isValidPermissionValue(
    poolValue: bigint,
    permissionValue: bigint,
  ): boolean {
    return (poolValue & permissionValue) === permissionValue;
  }
}
```

### 3.2 应用类型服务

```typescript
// src/modules/app-type/app-type.service.ts
import {
  Injectable,
  NotFoundException,
  BadRequestException,
} from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository, DataSource } from 'typeorm';
import { AppType } from '../entities/app-type.entity';
import { AppTypePermission } from '../entities/app-type-permission.entity';
import { Role } from '../entities/role.entity';

@Injectable()
export class AppTypeService {
  constructor(
    @InjectRepository(AppType)
    private appTypeRepository: Repository<AppType>,
    @InjectRepository(AppTypePermission)
    private appTypePermissionRepository: Repository<AppTypePermission>,
    @InjectRepository(Role)
    private roleRepository: Repository<Role>,
    private dataSource: DataSource,
  ) {}

  async findAll(page: number, pageSize: number, status?: number) {
    const queryBuilder = this.appTypeRepository.createQueryBuilder('appType');

    if (status !== undefined) {
      queryBuilder.where('appType.typeStatus = :status', { status });
    }

    const [list, total] = await queryBuilder
      .orderBy('appType.sortOrder', 'ASC')
      .skip((page - 1) * pageSize)
      .take(pageSize)
      .getManyAndCount();

    return { list, total };
  }

  async create(dto: CreateAppTypeDto) {
    // 检查 typeCode 是否已存在
    const existing = await this.appTypeRepository.findOne({
      where: { typeCode: dto.typeCode },
    });
    if (existing) {
      throw new BadRequestException('应用类型编码已存在');
    }

    const appType = this.appTypeRepository.create(dto);
    return this.appTypeRepository.save(appType);
  }

  async update(id: string, dto: UpdateAppTypeDto) {
    const appType = await this.appTypeRepository.findOne({ where: { id } });
    if (!appType) {
      throw new NotFoundException('应用类型不存在');
    }

    // typeCode 不允许修改
    if (dto.typeCode && dto.typeCode !== appType.typeCode) {
      throw new BadRequestException('应用类型编码不可修改');
    }

    Object.assign(appType, dto);
    return this.appTypeRepository.save(appType);
  }

  async delete(id: string) {
    const appType = await this.appTypeRepository.findOne({ where: { id } });
    if (!appType) {
      throw new NotFoundException('应用类型不存在');
    }

    // 系统内置类型不允许删除
    if (appType.typeCode === 'system') {
      throw new BadRequestException('系统内置类型不可删除');
    }

    // 检查是否有关联的应用实例
    const appCount = await this.dataSource
      .createQueryBuilder()
      .select('COUNT(*)', 'count')
      .from('sys_app', 'app')
      .where('app.appTypeId = :id', { id })
      .getRawOne();

    if (appCount.count > 0) {
      throw new BadRequestException('该应用类型下存在应用实例，无法删除');
    }

    // 软删除：更新状态
    appType.typeStatus = 0;
    await this.appTypeRepository.save(appType);
  }
}
```

### 3.3 权限池服务

```typescript
// src/modules/app-type/app-type-permission.service.ts
import { Injectable } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository, DataSource } from 'typeorm';
import { AppTypePermission } from '../entities/app-type-permission.entity';
import { Permission } from '../entities/permission.entity';
import { PermissionCalculationService } from '../permission/permission-calculation.service';

@Injectable()
export class AppTypePermissionService {
  constructor(
    @InjectRepository(AppTypePermission)
    private appTypePermissionRepository: Repository<AppTypePermission>,
    @InjectRepository(Permission)
    private permissionRepository: Repository<Permission>,
    private permissionCalcService: PermissionCalculationService,
    private dataSource: DataSource,
  ) {}

  async getPermissionsByAppTypeId(appTypeId: string) {
    return this.appTypePermissionRepository.find({
      where: { appTypeId },
      relations: ['permission'],
    });
  }

  async configurePermissions(
    appTypeId: string,
    permissions: Array<{ permissionId: string; permissionValue: bigint }>,
  ) {
    const queryRunner = this.dataSource.createQueryRunner();
    await queryRunner.connect();
    await queryRunner.startTransaction();

    try {
      // 删除旧的权限池配置
      await queryRunner.manager.delete(AppTypePermission, { appTypeId });

      // 验证每个权限值是否在 Permission.permissionValue 的子集范围内
      for (const item of permissions) {
        const permission = await queryRunner.manager.findOne(Permission, {
          where: { id: item.permissionId },
        });

        if (!permission) {
          throw new Error(`权限不存在：${item.permissionId}`);
        }

        // 验证：(poolValue & permValue) === poolValue
        if (!this.permissionCalcService.isValidPermissionValue(
          permission.permissionValue,
          item.permissionValue,
        )) {
          throw new Error(
            `权限值 ${item.permissionValue} 超出了权限 ${permission.permCode} 的范围`,
          );
        }

        // 创建新的权限池配置
        const appTypePermission = queryRunner.manager.create(AppTypePermission, {
          appTypeId,
          permissionId: item.permissionId,
          permissionValue: item.permissionValue,
        });

        await queryRunner.manager.save(appTypePermission);
      }

      await queryRunner.commitTransaction();
    } catch (error) {
      await queryRunner.rollbackTransaction();
      throw error;
    } finally {
      await queryRunner.release();
    }
  }
}
```

### 4.2 权限守卫实现

```typescript
// src/common/guards/permission.guard.ts
import {
  Injectable,
  CanActivate,
  ExecutionContext,
  ForbiddenException,
} from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import { REQUIRE_PERMISSION } from '../decorators/require-permission.decorator';
import { PermissionCalculationService } from '../../modules/permission/permission-calculation.service';

@Injectable()
export class PermissionGuard implements CanActivate {
  constructor(
    private reflector: Reflector,
    private permissionCalcService: PermissionCalculationService,
  ) {}

  async canActivate(context: ExecutionContext): Promise<boolean> {
    const requirePermission = this.reflector.get(
      REQUIRE_PERMISSION,
      context.getHandler(),
    );

    if (!requirePermission) {
      return true;
    }

    const { permCode, permissionValue = 0n } = requirePermission;
    const request = context.switchToHttp().getRequest();
    const user = request.user;

    if (!user) {
      throw new ForbiddenException('未认证');
    }

    // 获取用户的权限
    const userPermissions = await this.permissionCalcService.calculateUserPermissions(
      user.id,
      request.headers['x-app-type-code'],
    );

    // 查找对应的权限值
    const userPermValue = userPermissions.get(permCode) || 0n;

    // 验证权限
    if (!this.permissionCalcService.hasPermission(userPermValue, permissionValue)) {
      throw new ForbiddenException('权限不足');
    }

    return true;
  }
}
```

### 6.2 事务处理

```typescript
async updateWithTransaction(id: string, dto: UpdateDto) {
  const queryRunner = this.dataSource.createQueryRunner();
  await queryRunner.connect();
  await queryRunner.startTransaction();

  try {
    // 业务逻辑
    await queryRunner.manager.update(Entity, id, dto);

    await queryRunner.commitTransaction();
  } catch (error) {
    await queryRunner.rollbackTransaction();
    throw error;
  } finally {
    await queryRunner.release();
  }
}
```

### 6.3 错误处理

```typescript
// src/common/filters/http-exception.filter.ts
import { ExceptionFilter, Catch, ArgumentsHost, HttpException } from '@nestjs/common';

@Catch(HttpException)
export class HttpExceptionFilter implements ExceptionFilter {
  catch(exception: HttpException, host: ArgumentsHost) {
    const ctx = host.switchToHttp();
    const response = ctx.getResponse();
    const status = exception.getStatus();
    const exceptionResponse = exception.getResponse();

    response.status(status).json({
      code: status,
      data: null,
      message: typeof exceptionResponse === 'string'
        ? exceptionResponse
        : (exceptionResponse as any).message,
    });
  }
}
```
