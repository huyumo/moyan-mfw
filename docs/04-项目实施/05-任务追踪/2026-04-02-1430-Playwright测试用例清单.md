- 覆盖文档流程图中的每个操作节点
- 每个页面操作对应一个或多个测试场景
- 包含正向流程和异常流程

**测试环境**：
- baseURL: `http://127.0.0.1:5174`
- 测试账号：`test` / `Test@123`（普通用户）、`admin` / `Admin@123`（管理员）

export async function login(page, username: string, password: string) {
  await page.goto('/login');
  await page.fill('[name="username"]', username);
  await page.fill('[name="password"]', password);
  await page.click('button[type="submit"]');
  await page.waitForURL(/\/dashboard|\/app-select/);
}

export async function loginAsAdmin(page) {
  return login(page, 'admin', 'Admin@123');
}

export async function loginAsUser(page) {
  return login(page, 'test', 'Test@123');
}

export async function selectApp(page, appName: string) {
  await page.click(`text=${appName}`);
  await page.waitForLoadState('networkidle');
}
