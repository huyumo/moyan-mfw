- [ ] **2.2.1** 在 `MfwModal.vue` 写入 `<template>`:
```vue
<template>
  <el-dialog
    v-model="internalVisible"
    :title="title"
    :width="width"
    :show-close="showClose"
    :close-on-click-modal="closeOnClickModal"
    :close-on-press-escape="closeOnPressEscape"
    :before-close="beforeClose"
    @open="handleOpen"
    @close="handleClose"
  >
    <slot></slot>
    
    <template #footer v-if="$slots.footer || actions">
      <slot name="footer">
        <el-button v-if="showCancel" @click="handleCancel">取消</el-button>
        <el-button
          v-for="(action, index) in actions"
          :key="index"
          :type="action.type"
          @click="handleActionClick(action)"
        >
          {{ action.label }}
        </el-button>
        <el-button v-if="showConfirm" type="primary" @click="handleConfirm">
          确认
        </el-button>
      </slot>
    </template>
  </el-dialog>
</template>
```

#### 2.3 编写组件脚本
- [ ] **2.3.1** 在 `MfwModal.vue` 写入 `<script setup lang="ts">`:
```typescript
import { defineOptions, computed } from 'vue'
import type { MfwModalProps, MfwModalEmits, MfwModalAction } from './MfwModal.types'

defineOptions({
  name: 'MfwModal',
  inheritAttrs: true
})

const props = withDefaults(defineProps<MfwModalProps>(), {
  title: '',
  width: '50%',
  showClose: true,
  closeOnClickModal: true,
  closeOnPressEscape: true,
  showCancel: true,
  showConfirm: false
})

const emit = defineEmits<MfwModalEmits>()

const internalVisible = defineModel<boolean>()

const actions = ref<MfwModalAction[]>([])

const handleOpen = () => {
  emit('open')
}

const handleClose = () => {
  emit('close')
}

const handleCancel = () => {
  internalVisible.value = false
}

const handleConfirm = () => {
  internalVisible.value = false
}

const handleActionClick = (action: MfwModalAction) => {
  action.onClick?.()
  emit('action-click', action.label)
}
```
