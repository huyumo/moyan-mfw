  <el-card class="mfw-form-card" :class="[`mfw-form-card--${size}`]">
    <template #header v-if="$slots.header || title">
      <slot name="header">
        <span class="mfw-form-card__title">{{ title }}</span>
      </slot>
    </template>
    
    <el-form
      :model="model"
      :rules="rules"
      :label-width="labelWidth"
      :disabled="disabled"
      @submit.prevent="handleSubmit"
      @validate="handleValidate"
    >
      <slot></slot>
    </el-form>
    
    <template #footer v-if="$slots.actions">
      <slot name="actions"></slot>
    </template>
  </el-card>
</template>
```

#### 2.3 编写组件脚本
- [ ] **2.3.1** 在 `MfwFormCard.vue` 写入 `<script setup lang="ts">`:
```typescript
import { defineOptions, defineModel } from 'vue'
import type { MfwFormCardProps, MfwFormCardEmits } from './MfwFormCard.types'

defineOptions({
  name: 'MfwFormCard',
  inheritAttrs: true
})

const props = withDefaults(defineProps<MfwFormCardProps>(), {
  labelWidth: '100px',
  size: 'default',
  disabled: false
})

const emit = defineEmits<MfwFormCardEmits>()

const handleSubmit = () => {
  emit('submit', props.model)
}

const handleValidate = (field: string, valid: boolean) => {
  emit('validate', field, valid)
}

const resetFields = () => {
  emit('reset')
}

defineExpose({
  resetFields
})
```

# MfwFormCard 组件

## 基础用法

```vue
<template>
  <MfwFormCard
    v-model="formData"
    :rules="rules"
    title="用户信息"
    @submit="handleSubmit"
  >
    <el-form-item label="姓名" prop="name">
      <el-input v-model="formData.name" />
    </el-form-item>
  </MfwFormCard>
</template>
```

## Props

| 参数 | 类型 | 默认值 | 说明 |
|------|------|--------|------|
| model | object | - | 表单数据模型 |
| rules | object | - | 验证规则 |
| labelWidth | string | 100px | 标签宽度 |
| size | string | default | 尺寸 |
| disabled | boolean | false | 是否禁用 |

## Events

| 事件名 | 回调参数 | 说明 |
|--------|----------|------|
| submit | (values) | 提交表单 |
| validate | (field, valid) | 验证结果 |
| reset | - | 重置表单 |

## 方法

| 方法名 | 参数 | 说明 |
|--------|------|------|
| validate | - | 验证整个表单 |
| validateField | field | 验证单个字段 |
| resetFields | - | 重置表单 |
```
