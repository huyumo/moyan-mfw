#### 2.3 编写组件脚本
- [ ] **2.3.1** 在 `MfwUpload.vue` 写入 `<script setup lang="ts">`:
```typescript
import { defineOptions, ref, watch } from 'vue'
import type { MfwUploadProps, MfwUploadEmits } from './MfwUpload.types'

defineOptions({
  name: 'MfwUpload',
  inheritAttrs: true
})

const props = withDefaults(defineProps<MfwUploadProps>(), {
  multiple: false,
  limit: 0,
  autoUpload: true,
  showFileList: true,
  drag: false
})

const emit = defineEmits<MfwUploadEmits>()

const uploadRef = ref<any>()
const internalFileList = ref<any[]>(props.fileList || [])

watch(() => props.fileList, (newVal) => {
  if (newVal) {
    internalFileList.value = newVal
  }
})

const handleChange = (file: any, fileList: any[]) => {
  internalFileList.value = fileList
  emit('change', file, fileList)
}

const handleSuccess = (response: any, file: any) => {
  emit('success', response, file)
}

const handleError = (error: any, file: any) => {
  emit('error', error, file)
}

const handleProgress = (event: ProgressEvent) => {
  emit('progress', event)
}

const handleRemove = (file: any) => {
  emit('remove', file)
}

const handleRemoveFile = (file: any) => {
  uploadRef.value?.remove(file)
}

const handlePreview = (file: any) => {
  window.open(file.url || URL.createObjectURL(file.raw))
}

const submit = () => {
  uploadRef.value?.submit()
}

defineExpose({
  submit,
  clearFiles: () => uploadRef.value?.clearFiles()
})
```

# MfwUpload 组件

## 基础用法

```vue
<template>
  <MfwUpload
    action="/api/upload"
    :multiple="true"
    :limit="5"
    @change="handleChange"
    @success="handleSuccess"
  >
    <el-button type="primary">选择文件</el-button>
  </MfwUpload>
</template>

<script setup lang="ts">
const handleChange = (file: any, fileList: any[]) => {
  console.log('文件变化', file)
}

const handleSuccess = (response: any, file: any) => {
  console.log('上传成功', response)
}
</script>
```

## 拖拽上传

```vue
<MfwUpload
  action="/api/upload"
  :drag="true"
  @success="handleSuccess"
/>
```

## Props

| 参数 | 类型 | 默认值 | 说明 |
|------|------|--------|------|
| action | string | - | 上传地址 |
| multiple | boolean | false | 是否多选 |
| limit | number | 0 | 最大上传数量 |
| accept | string | - | 接受的文件类型 |
| autoUpload | boolean | true | 是否自动上传 |
| drag | boolean | false | 是否支持拖拽 |

## Events

| 事件名 | 回调参数 | 说明 |
|--------|----------|------|
| change | (file, fileList) | 文件列表变化 |
| success | (response, file) | 上传成功 |
| error | (error, file) | 上传失败 |
| progress | (event) | 上传进度 |
| remove | (file) | 删除文件 |

## 方法

| 方法名 | 参数 | 说明 |
|--------|------|------|
| submit | - | 手动提交上传 |
| clearFiles | - | 清空文件列表 |

## 插槽

| 名称 | 说明 |
|------|------|
| default | 触发按钮内容 |
| tip | 提示文字 |
```
