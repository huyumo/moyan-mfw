```vue
    :closable="closable"
    :addable="addable"
    @tab-click="handleTabClick"
    @tab-change="handleTabChange"
    @tab-remove="handleTabRemove"
    @tab-add="handleTabAdd"
  >
    <slot></slot>
  </el-tabs>
</template>
```
- [ ] **2.2.2** 保存文件

#### 2.3 编写 Tabs 组件脚本
- [ ] **2.3.1** 在 `MfwTabs.vue` 写入 `<script setup lang="ts">`:
```typescript
import { defineOptions } from 'vue'
import type { MfwTabsProps, MfwTabsEmits } from './MfwTabs.types'

defineOptions({
  name: 'MfwTabs',
  inheritAttrs: true
})

const props = withDefaults(defineProps<MfwTabsProps>(), {
  type: 'line',
  tabPosition: 'top',
  closable: false,
  addable: false
})

const emit = defineEmits<MfwTabsEmits>()

const internalValue = defineModel<string | number>()

const handleTabClick = (tab: any) => {
  emit('tab-click', tab)
}

const handleTabChange = (value: string | number) => {
  emit('tab-change', value)
}

const handleTabRemove = (name: string | number) => {
  emit('tab-remove', name)
}

const handleTabAdd = () => {
  emit('tab-add')
}
```
- [ ] **2.3.2** 保存文件
