- [ ] **2.2.1** 在 `MfwPicker.vue` 写入 `<template>`:

```vue
<template>
  <el-select
    v-model="internalValue"
    :placeholder="placeholder"
    :multiple="multiple"
    :filterable="filterable"
    :clearable="clearable"
    :disabled="disabled"
    :size="size"
    @change="handleChange"
    @focus="handleFocus"
    @blur="handleBlur"
    @clear="handleClear"
  >
    <el-option
      v-for="option in options"
      :key="option.value"
      :label="option.label"
      :value="option.value"
      :disabled="option.disabled"
    />
  </el-select>
</template>
```
- [ ] **2.2.2** 保存文件

#### 2.3 编写组件脚本
- [ ] **2.3.1** 在 `MfwPicker.vue` 写入 `<script setup lang="ts">`:
```typescript
import { defineOptions } from 'vue'
import type { MfwPickerProps, MfwPickerEmits } from './MfwPicker.types'

defineOptions({
  name: 'MfwPicker',
  inheritAttrs: true
})

const props = withDefaults(defineProps<MfwPickerProps>(), {
  placeholder: '请选择',
  multiple: false,
  filterable: false,
  clearable: false,
  disabled: false,
  size: 'default'
})

const emit = defineEmits<MfwPickerEmits>()

const internalValue = defineModel<any>()

const handleChange = (value: any) => {
  emit('change', value)
}

const handleFocus = (event: FocusEvent) => {
  emit('focus', event)
}

const handleBlur = (event: FocusEvent) => {
  emit('blur', event)
}

const handleClear = () => {
  emit('clear')
}
```
