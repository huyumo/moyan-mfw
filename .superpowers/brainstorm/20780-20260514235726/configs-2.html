📄 tsconfig 配置 + 扩展包 + 导入转换流程

5. packages/base/src/backend/tsconfig.json

后端独立 tsconfig — 只管自己，不再继承 shared
{
"extends": "../../../../tsconfig.base.json",
"compilerOptions": {
"module": "commonjs",
"moduleResolution": "node",
"strictPropertyInitialization": false,
"composite": true,
"outDir": "../../dist/backend",
"declarationMap": true,
"baseUrl": ".",
"paths": {
"@/*": ["./*"],
"@common/*": ["common/*"],
"@modules/*": ["modules/*"]
}
},
"include": ["./**/*.ts"],
"exclude": ["node_modules"]
}

6. packages/base/src/shared/tsconfig.json

共享层完全独立 — 不再被迫继承 backend 的 CJS
{
"extends": "../../../../tsconfig.base.json",
"compilerOptions": {
"module": "commonjs", ← backend 消费 CJS
"moduleResolution": "node",
"composite": true,
"outDir": "../../dist/shared",
"declarationMap": true
},
"include": ["./**/*.ts"]
}

8. packages/extensions/extension-ad/src/backend/package.json

扩展包后端 — 通过 @internal 引用 base
{
"name": "@internal/ad-backend",
"private": true,

"main": "./index.ts",
"types": "./index.ts",

"scripts": {
"build": "tsc && tsc-alias",
"dev": "nest start --watch"

"dependencies": {

"@internal/base-shared": "workspace:*",

"@nestjs/common": "catalog:",

},

"tsc-alias": "catalog:",

}
}

9. tsconfig.publish.json（npm 发布前转换）

build:publish 脚本用此配置，tsc-alias 将 @internal 转为 npm 路径
{
// packages/base/tsconfig.publish.json
"extends": "./tsconfig.json",
"compilerOptions": {
"paths": {
"@internal/base-shared": ["./dist/shared/index.js"],
"@internal/base-backend": ["./dist/backend/index.js"],
"@internal/base-frontend": ["./dist/frontend/index.js"]
}
},
"include": ["./dist/**/*.js"]
}
packages/extensions/extension-ad/tsconfig.publish.json
{
// 扩展包的 @internal → 消费者可解析的 npm 路径
"extends": "./tsconfig.json",
"compilerOptions": {
"paths": {
"@internal/base-backend": ["moyan-mfw-base/backend"],
"@internal/base-shared": ["moyan-mfw-base/shared"],
"@internal/base-frontend": ["moyan-mfw-base/frontend"],
"@internal/ad-shared": ["../shared/index.js"] ← 内部引用转相对路径
}
},
"include": ["./dist/**/*.js"]
}

🔄 导入路径转换流程

阶段 1：源码（dev）

// extension-ad/src/backend/controller/ad.controller.ts
import { AuthGuard } from '@internal/base-backend';
import { StatusDict } from '@internal/base-shared';
import { LINK_TYPE } from '@internal/ad-shared';

├─ dev 解析：pnpm workspace 软链接 → node_modules/@internal/base-backend → 标准 JS 解析

阶段 2：tsc 编译

tsc 按 tsconfig 编译为 CJS，output 中 保留 require('@internal/base-backend')

阶段 3：tsc-alias 转换（prepublishOnly）

// dist/backend/controller/ad.controller.js（转换后）
const { AuthGuard } = require('moyan-mfw-base/backend');
const { StatusDict } = require('moyan-mfw-base/shared');
const { LINK_TYPE } = require('../shared/index.js'); ← 自引用转相对

阶段 4：消费者安装

npm i moyan-mfw-extension-ad moyan-mfw-base
Node.js 通过 moyan-mfw-base 的 exports 字段解析 → 标准 npm 行为 ✅
