📄 关键配置文件详细设计

1. pnpm-workspace.yaml 变更

注册所有子包路径
packages: - packages/base/src/backend ← @internal/base-backend - packages/base/src/frontend ← @internal/base-frontend - packages/base/src/shared ← @internal/base-shared - packages/base ← moyan-mfw-base (根聚合包) - packages/extensions/*/src/backend ← @internal/ad-backend 等 - packages/extensions/*/src/frontend - packages/extensions/*/src/shared - packages/extensions/* ← 扩展根聚合包 - packages/component-demo - ...

2. packages/base/package.json（npm 发布入口）

{
"name": "moyan-mfw-base",
"version": "1.0.0",
"description": "墨焱MFW核心框架",

"exports": {
"./backend": {
"require": "./dist/backend/index.js",
"types": "./dist/backend/index.d.ts"
},
"./frontend": {
"import": "./dist/frontend/index.mjs",
"require": "./dist/frontend/index.js",
"types": "./dist/frontend/index.d.ts"
},
"./shared": {
"types": "./dist/shared/index.d.ts",
"require": "./dist/shared/index.js",
"default": "./dist/shared/index.js"
}
},

"scripts": {
"build": "pnpm run build:shared && pnpm run build:backend && pnpm run build:frontend",
"build:backend": "pnpm --filter @internal/base-backend build",
"build:frontend": "pnpm --filter @internal/base-frontend build",
"build:shared": "pnpm --filter @internal/base-shared build",
"build:publish": "pnpm run build && tsc-alias -p tsconfig.publish.json",
"prepublishOnly": "pnpm run build:publish"
},

"files": [
"dist/",
"migrations/"
],

"devDependencies": {
"tsc-alias": "catalog:"
}
}
