## 9.1 单元测试

### 9.1.1 测试框架

项目使用 **Vitest** 作为单元测试框架：

```bash
# 运行测试
pnpm run test:unit

# 监听模式
pnpm run test:unit:watch
```

### 9.1.2 测试文件命名

```
✅ 推荐：
ComponentName.test.ts
ComponentName.spec.ts
useComposable.test.ts

❌ 避免：
test.ts
ComponentName-test.ts
```

### 9.1.3 测试结构

```typescript
import { describe, it, expect, vi } from 'vitest';
import { ref } from 'vue';
import { useUserList } from './useUserList';

describe('useUserList', () => {
  it('应该初始化空列表', () => {
    const { list } = useUserList();
    expect(list.value).toEqual([]);
  });

  it('应该加载数据', async () => {
    const { list, loadData } = useUserList();
    await loadData();
    expect(list.value.length).toBeGreaterThan(0);
  });

  it('应该处理加载错误', async () => {
    const { loading, loadData } = useUserList();
    // 模拟错误
    await expect(loadData()).rejects.toThrow();
    expect(loading.value).toBe(false);
  });
});
```

### 9.1.4 组件测试

```typescript
import { mount } from '@vue/test-utils';
import { describe, it, expect } from 'vitest';
import UserCard from './UserCard.vue';

describe('UserCard', () => {
  it('应该渲染用户名', () => {
    const wrapper = mount(UserCard, {
      props: {
        userName: '张三',
      },
    });
    expect(wrapper.text()).toContain('张三');
  });

  it('应该触发点击事件', async () => {
    const wrapper = mount(UserCard, {
      props: {
        userName: '张三',
      },
    });
    await wrapper.find('button').trigger('click');
    expect(wrapper.emitted('click')).toBeTruthy();
  });
});
```

## 9.2 E2E 测试

### 9.2.1 测试框架

项目使用 **Playwright** 进行 E2E 测试：

```bash
# 运行 E2E 测试
pnpm run test:e2e

# 有头模式（显示浏览器）
pnpm run test:e2e:ui
```

### 9.2.2 测试结构

```typescript
import { test, expect } from '@playwright/test';

test.describe('登录流程', () => {
  test('应该成功登录', async ({ page }) => {
    await page.goto('/login');

    // 填写表单
    await page.fill('[name="username"]', 'admin');
    await page.fill('[name="password"]', 'password');

    // 提交表单
    await page.click('button[type="submit"]');

    // 验证跳转
    await expect(page).toHaveURL('/dashboard');
  });

  test('应该显示错误信息', async ({ page }) => {
    await page.goto('/login');

    // 填写错误密码
    await page.fill('[name="username"]', 'admin');
    await page.fill('[name="password"]', 'wrong');
    await page.click('button[type="submit"]');

    // 验证错误信息
    await expect(page.locator('.error-message')).toBeVisible();
  });
});
```

### 9.2.3 Page Object 模式

```typescript
// pages/login.page.ts
export class LoginPage {
  constructor(private page) {}

  async goto() {
    await this.page.goto('/login');
  }

  async login(username: string, password: string) {
    await this.page.fill('[name="username"]', username);
    await this.page.fill('[name="password"]', password);
    await this.page.click('button[type="submit"]');
  }
}

// tests/login.spec.ts
import { test, expect } from '@playwright/test';
import { LoginPage } from './pages/login.page';

test.describe('登录', () => {
  let loginPage: LoginPage;

  test.beforeEach(async ({ page }) => {
    loginPage = new LoginPage(page);
  });

  test('成功登录', async ({ page }) => {
    await loginPage.goto();
    await loginPage.login('admin', 'password');
    await expect(page).toHaveURL('/dashboard');
  });
});
```
